#!/usr/bin/env bash
# Phase 28-05 (POL-06) — alembic migration round-trip safety check.
# Usage: scripts/alembic-roundtrip.sh
# Requires: docker-compose stack running (`docker compose up -d db api`).
set -euo pipefail

DOCKER_COMPOSE="${DOCKER_COMPOSE:-docker compose}"
API_SERVICE="${API_SERVICE:-api}"

echo "=== Phase 28-05 alembic round-trip ==="
echo "Step 1/3: alembic upgrade head"
$DOCKER_COMPOSE exec -T "$API_SERVICE" alembic upgrade head

echo "Step 2/3: alembic downgrade -1"
$DOCKER_COMPOSE exec -T "$API_SERVICE" alembic downgrade -1

echo "Step 3/3: alembic upgrade head (re-apply)"
$DOCKER_COMPOSE exec -T "$API_SERVICE" alembic upgrade head

echo "Round-trip OK"
```

/**
 * Phase 28-05 (POL-07) — §14 ТЗ acceptance happy-path.
 *
 * Один длинный test покрывает весь user-journey:
 *   1. Onboarding 4-step submit (income → accounts → plan → goal/skip → final)
 *   2. Home renders «Дневной темп» с count-up
 *   3. FAB → AddSheet → ввести сумму через keypad → выбрать категорию + счёт → СОХРАНИТЬ → toast
 *   4. Push to Transactions → запись видна
 *   5. Push to PLAN → меняем slider → СОХРАНИТЬ
 *   6. Tab → AI → видим observation + 4 chips
 *   7. Tab → Savings → видим plate + цели
 *
 * Wall-clock budget: < 60s (per ТЗ §14.1).
 */
import { test, expect } from '@playwright/test';

test.use({ actionTimeout: 8000 });

test('§14 ТЗ acceptance happy-path', async ({ page }) => {
  test.setTimeout(60_000); // ТЗ §14.1: онбординг + smoke < 60s wall-clock

  await page.addInitScript(() => localStorage.setItem('ui.theme', 'v10'));

  const start = Date.now();

  // ─── 1. Onboarding ─────────────────────────────────────────────
  await page.goto('/');

  // Step 01: income (NEXT enabled at income > 0)
  // Селекторы: смотри onboarding-v10.spec.ts для проверенных testid; адаптируй здесь.
  await page.getByRole('textbox').first().fill('120000');  // 120 000 ₽
  await page.getByRole('button', { name: /ДАЛЕЕ|NEXT/i }).click();

  // Step 02: добавить хотя бы один счёт
  await page.getByRole('button', { name: /Т-Банк|Сбер|Наличные|Добавить/i }).first().click();
  // Если открывается balance form — заполни:
  const balanceField = page.getByRole('textbox').last();
  if (await balanceField.count() > 0) {
    await balanceField.fill('50000');
    const sheetSubmit = page.getByRole('button', { name: /ГОТОВО|ОК|ДОБАВИТЬ/i });
    if (await sheetSubmit.count() > 0) await sheetSubmit.first().click();
  }
  await page.getByRole('button', { name: /ДАЛЕЕ|NEXT/i }).click();

  // Step 03: plan distribution — accept defaults via NEXT
  await page.getByRole('button', { name: /ДАЛЕЕ|NEXT/i }).click();

  // Step 04: skip goal
  const skipBtn = page.getByRole('button', { name: /ПРОПУСТИТЬ|SKIP/i });
  if (await skipBtn.count() > 0) await skipBtn.first().click();

  // Final: НАЧАТЬ →
  await page.getByRole('button', { name: /НАЧАТЬ|START/i }).click();

  // ─── 2. Home renders «Дневной темп» с count-up ─────────────────
  await expect(page.getByText(/Дневной темп/i)).toBeVisible({ timeout: 6000 });
  await expect(page.locator('.poster-rise-in').first()).toBeVisible();

  // ─── 3. FAB → AddSheet → save ──────────────────────────────────
  const fab = page.locator('[aria-label*="Добавить" i], [data-testid="fab"]').first();
  await fab.click();
  await expect(page.getByText(/NEW ENTRY/i)).toBeVisible({ timeout: 4000 });

  // Введи сумму через keypad: тап на «1», «0», «0», «0»
  for (const k of ['1', '0', '0', '0']) {
    await page.getByRole('button', { name: new RegExp(`^${k}$`) }).first().click();
  }

  // Выбери первую категорию
  await page.locator('[data-testid^="cat-chip-"], .catChip').first().click();

  // СОХРАНИТЬ
  await page.getByRole('button', { name: /СОХРАНИТЬ/i }).click();

  // Toast «✓ ДОБАВЛЕНО» / «−10 ₽ → реестр» (любая form'ы)
  const toast = page.locator('.poster-toast-in, [role="status"]').first();
  await toast.waitFor({ state: 'visible', timeout: 4000 }).catch(() => {});

  // ─── 4. Transactions: push from Home «ВСЕ ОПЕРАЦИИ» → запись видна ─
  await page.getByText(/ВСЕ ОПЕРАЦИИ/i).first().click();
  await expect(page.getByText(/Реестр/i)).toBeVisible({ timeout: 4000 });
  // Запись на 10 ₽ должна быть в списке (если backend mock-friendly или live).

  // ─── 5. PLAN мая push + edit + save ────────────────────────────
  // Возврат на Home (back-arrow или tab)
  await page.goBack();
  await page.getByText(/PLAN МЕСЯЦА/i).first().click();
  await expect(page.getByText(/PLAN МЕСЯЦА|ОСТАЛОСЬ РАСПРЕДЕЛИТЬ/i)).toBeVisible();

  // Найди первый slider, увеличь на step (через keyboard arrow если sliders accessible)
  const slider = page.locator('[role="slider"], input[type="range"]').first();
  if (await slider.count() > 0) {
    await slider.focus();
    await slider.press('ArrowRight');
  }
  const saveBtn = page.getByRole('button', { name: /СОХРАНИТЬ/i });
  if (await saveBtn.isEnabled()) await saveBtn.click();

  // ─── 6. AI tab — initial observation + 4 chips ─────────────────
  await page.goBack();
  await page.locator('[data-testid="tab-ai"], [aria-label*="AI" i]').first().click();
  await expect(page.getByText(/ASSISTANT|ONLINE/i)).toBeVisible({ timeout: 4000 });
  // 4 chip-suggestions с DM Serif italic 18px — locator любой class .chipSuggestion.
  const chipCount = await page.locator('[data-testid^="ai-chip-"], .chipSuggestion').count();
  expect(chipCount).toBeGreaterThanOrEqual(4);

  // ─── 7. Savings tab — plate + goals ────────────────────────────
  await page.locator('[data-testid="tab-savings"], [aria-label*="Копилка" i]').first().click();
  await expect(page.getByText(/Копилка|НАКОПЛЕНО/i)).toBeVisible({ timeout: 4000 });

  const elapsed = Date.now() - start;
  console.log(`§14 happy-path elapsed: ${elapsed}ms`);
  expect(elapsed).toBeLessThan(60_000); // ТЗ §14.1
});

test.skip('§14.7 no visible FOUT after first visit (manual smoke)', async () => {
  // Cannot programmatically detect "no FOUT" reliably от Playwright.
  // Проверяется manually на TG Mini App пост-deploy. Marked .skip() как documentation.
});
```
