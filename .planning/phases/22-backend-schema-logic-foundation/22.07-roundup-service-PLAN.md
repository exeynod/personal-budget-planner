```python
"""Roundup service (BE-07).

Service-layer Python hook called from app/services/actual.create_actual after
the parent expense transaction is inserted. Creates a child kind='roundup'
transaction in the same DB transaction; updates account.balance_cents
atomically.

Formula (DATA-MODEL §4):
    delta = ((|amount| + base - 1) // base) * base - |amount|

Skip cases (CONTEXT §Area 3):
    - delta == 0   (amount already a multiple of base)
    - delta == base (treats edge — amount was exactly at boundary; ceil(0/base)*base = 0)
    - parent.kind != 'expense'
    - SavingsConfig absent or roundup_enabled=False

Roundup is opt-in: SavingsConfig is created at first onboarding-complete (plan 22.11)
with default roundup_enabled=False. Toggle via PATCH /api/v1/savings/config.
"""
from __future__ import annotations

from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.db.models import (
    ActualKind, ActualSource, ActualTransaction, Category, SavingsConfig
)


class SavingsCategoryMissingError(Exception):
    """Raised when no Category with code='savings' exists for user (BE-07).

    Onboarding-complete (plan 22.11) is responsible for seeding this row.
    Surfaces as 500 to caller — config drift, not user error.
    """
    def __init__(self, user_id: int) -> None:
        self.user_id = user_id
        super().__init__(
            f"No savings system Category (code='savings') exists for user_id={user_id}; "
            "onboarding may not have completed correctly"
        )


def compute_roundup_delta(amount_cents: int, base: int) -> int:
    """Pure function — compute roundup delta for given amount + base.

    Returns:
        Non-negative integer delta (0 means no roundup needed).
    """
    abs_amt = abs(amount_cents)
    if abs_amt == 0:
        return 0
    rounded_up = ((abs_amt + base - 1) // base) * base
    return rounded_up - abs_amt


def should_skip(delta: int, base: int) -> bool:
    """CONTEXT §Area 3 skip rules: delta == 0 OR delta == base."""
    return delta == 0 or delta == base


async def get_savings_config(db: AsyncSession, *, user_id: int) -> SavingsConfig | None:
    return await db.scalar(
        select(SavingsConfig).where(SavingsConfig.user_id == user_id)
    )


async def get_savings_category(db: AsyncSession, *, user_id: int) -> Category | None:
    """Resolve the system 'savings' Category for user (code='savings')."""
    return await db.scalar(
        select(Category).where(
            Category.user_id == user_id,
            Category.code == "savings",
        )
    )


async def maybe_create_roundup_child(
    db: AsyncSession,
    *,
    user_id: int,
    parent_txn: ActualTransaction,
) -> ActualTransaction | None:
    """Create roundup child if conditions met. Idempotent: caller must not call twice for same parent.

    Returns child txn or None.

    Side effects:
        - Inserts child ActualTransaction (no commit — caller's transaction).
        - Updates parent.account_id account.balance_cents via apply_balance_delta.
    """
    # Trigger gate: only kind='expense' parents trigger roundup.
    if parent_txn.kind != ActualKind.expense:
        return None

    cfg = await get_savings_config(db, user_id=user_id)
    if cfg is None or not cfg.roundup_enabled:
        return None

    delta = compute_roundup_delta(parent_txn.amount_cents, cfg.roundup_base)
    if should_skip(delta, cfg.roundup_base):
        return None

    savings_cat = await get_savings_category(db, user_id=user_id)
    if savings_cat is None:
        raise SavingsCategoryMissingError(user_id)

    # Insert child txn. Negative amount mirrors expense convention.
    child = ActualTransaction(
        user_id=user_id,
        period_id=parent_txn.period_id,
        kind=ActualKind.roundup,
        amount_cents=-delta,
        description="Округление",
        category_id=savings_cat.id,
        tx_date=parent_txn.tx_date,
        source=parent_txn.source,
        parent_txn_id=parent_txn.id,
        account_id=parent_txn.account_id,
    )
    db.add(child)
    await db.flush()
    await db.refresh(child)

    # Update account balance for the roundup amount.
    if parent_txn.account_id is not None:
        # Local import to avoid circular dependency at module import time.
        from app.services.accounts import apply_balance_delta
        await apply_balance_delta(
            db, parent_txn.account_id,
            user_id=user_id,
            delta_cents=-delta,
        )

    return child
```

```python
async def delete_actual_v10(
    db: AsyncSession, actual_id: int, *, user_id: int
) -> ActualTransaction:
    """v1.0 delete that restores account balance for both parent + cascading child.

    DB-level FK ON DELETE CASCADE handles child txn removal automatically.
    Service-layer aggregates pre-delete amounts to restore balance.
    """
    row = await get_or_404(db, actual_id, user_id=user_id)

    # Sum parent + any child txns BEFORE delete
    children = (await db.execute(
        select(ActualTransaction).where(
            ActualTransaction.parent_txn_id == row.id,
            ActualTransaction.user_id == user_id,
        )
    )).scalars().all()
    total_to_restore = -row.amount_cents - sum(c.amount_cents for c in children)
    affected_accounts: dict[int, int] = {}
    if row.account_id is not None:
        affected_accounts[row.account_id] = affected_accounts.get(row.account_id, 0) + (-row.amount_cents)
    for c in children:
        if c.account_id is not None:
            affected_accounts[c.account_id] = affected_accounts.get(c.account_id, 0) + (-c.amount_cents)

    await db.delete(row)  # CASCADE removes children
    await db.flush()

    # Restore balances per account
    from app.services.accounts import apply_balance_delta
    for acct_id, delta in affected_accounts.items():
        await apply_balance_delta(db, acct_id, user_id=user_id, delta_cents=delta)

    return row
```
