**Module-level transliteration map (covers existing seed names + safe latin passthrough):**
```python
# Best-effort cyrillic→latin transliteration map — covers v0.x seed names.
# CONTEXT §Area 1: collision handling = append '-2', '-3' suffix.
_TRANSLIT = {
    "а": "a", "б": "b", "в": "v", "г": "g", "д": "d", "е": "e", "ё": "yo",
    "ж": "zh", "з": "z", "и": "i", "й": "y", "к": "k", "л": "l", "м": "m",
    "н": "n", "о": "o", "п": "p", "р": "r", "с": "s", "т": "t", "у": "u",
    "ф": "f", "х": "h", "ц": "ts", "ч": "ch", "ш": "sh", "щ": "sch",
    "ъ": "", "ы": "y", "ь": "", "э": "e", "ю": "yu", "я": "ya",
}

def _transliterate(name: str) -> str:
    """Lowercase + cyrillic→latin + replace non-alphanumerics with '_'."""
    out = []
    for ch in name.lower():
        if ch in _TRANSLIT:
            out.append(_TRANSLIT[ch])
        elif ch.isalnum():
            out.append(ch)
        else:
            out.append("_")
    s = "".join(out).strip("_")
    return s or "cat"
```

# 2d. Backfill code via Python with collision handling per (user_id).
rows = conn.execute(sa.text("SELECT id, user_id, name FROM category ORDER BY user_id, id")).fetchall()
seen: dict[tuple[int, str], int] = {}  # (user_id, code) → count
for row in rows:
    base = _transliterate(row.name)
    key = (row.user_id, base)
    seen[key] = seen.get(key, 0) + 1
    code = base if seen[key] == 1 else f"{base}-{seen[key]}"
    conn.execute(
        sa.text("UPDATE category SET code = :code WHERE id = :id"),
        {"code": code, "id": row.id},
    )
