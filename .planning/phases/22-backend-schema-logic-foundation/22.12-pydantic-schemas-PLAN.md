from datetime import datetime

    mask: Optional[str] = Field(default=None, max_length=16)
    kind: Optional[Literal["card", "cash", "savings"]] = None
    balance_cents: Optional[int] = Field(default=None, ge=-100_000_000_00, le=100_000_000_00)

"""Pydantic v2 schemas for /api/v1/goals (BE-11)."""
from datetime import date as date_type, datetime
from typing import Optional

from pydantic import BaseModel, ConfigDict, Field, field_validator


class GoalCreate(BaseModel):
    model_config = ConfigDict(strict=True, extra="forbid", str_strip_whitespace=True)
    name: str = Field(min_length=1, max_length=80)
    target_cents: int = Field(gt=0, le=100_000_000_00)
    due: Optional[date_type] = None

    @field_validator("due")
    @classmethod
    def _due_in_future(cls, v: Optional[date_type]) -> Optional[date_type]:
        if v is None:
            return v
        from datetime import date as _d
        if v <= _d.today():
            raise ValueError("Goal due must be in the future")
        return v


class GoalUpdate(BaseModel):
    model_config = ConfigDict(strict=True, extra="forbid")
    name: Optional[str] = Field(default=None, min_length=1, max_length=80)
    target_cents: Optional[int] = Field(default=None, gt=0, le=100_000_000_00)
    due: Optional[date_type] = None


class GoalRead(BaseModel):
    model_config = ConfigDict(from_attributes=True)
    id: int
    name: str
    target_cents: int
    current_cents: int
    due: Optional[date_type]
    created_at: datetime
```

"""Pydantic v2 schemas for POST /api/v1/onboarding/complete v1.0 (BE-15)."""
from datetime import date as date_type
from typing import Literal, Optional
from pydantic import BaseModel, ConfigDict, Field, field_validator, model_validator


INCOME_MAX_CENTS = 100_000_000_00  # 100M ₽

VALID_DEFAULT_CODES = frozenset({
    "food", "cafe", "home", "transit", "fun", "gifts", "health", "subs"
})


class OnboardingAccountItem(BaseModel):
    model_config = ConfigDict(strict=True, extra="forbid", str_strip_whitespace=True)
    bank: str = Field(min_length=1, max_length=40)
    mask: Optional[str] = Field(default=None, max_length=16)
    kind: Literal["card", "cash", "savings"]
    balance_cents: int = Field(default=0, ge=-100_000_000_00, le=100_000_000_00)
    primary: bool = False


class OnboardingGoalItem(BaseModel):
    model_config = ConfigDict(strict=True, extra="forbid", str_strip_whitespace=True)
    name: str = Field(min_length=1, max_length=80)
    target_cents: int = Field(gt=0, le=INCOME_MAX_CENTS)
    due: Optional[date_type] = None


class OnboardingSavingsConfigItem(BaseModel):
    model_config = ConfigDict(strict=True, extra="forbid")
    roundup_enabled: bool = False
    base: Literal[10, 50, 100] = 10


class OnboardingV10Body(BaseModel):
    model_config = ConfigDict(strict=True, extra="forbid")
    income_cents: int = Field(gt=0, le=INCOME_MAX_CENTS)
    accounts: list[OnboardingAccountItem] = Field(min_length=1, max_length=10)
    category_plans: dict[str, int]  # code -> cents
    goal: Optional[OnboardingGoalItem] = None
    savings_config: Optional[OnboardingSavingsConfigItem] = None

    @field_validator("category_plans")
    @classmethod
    def _validate_codes(cls, v: dict[str, int]) -> dict[str, int]:
        for code, cents in v.items():
            if code not in VALID_DEFAULT_CODES:
                raise ValueError(f"Unknown category code: {code!r}")
            if not isinstance(cents, int) or cents < 0:
                raise ValueError(f"category_plans[{code!r}] must be non-negative int")
        return v

    @model_validator(mode="after")
    def _sum_plan_le_income(self) -> "OnboardingV10Body":
        sum_plan = sum(self.category_plans.values())
        if sum_plan > self.income_cents:
            raise ValueError(
                f"Sum of category_plans ({sum_plan}) exceeds income_cents ({self.income_cents})"
            )
        # Each plan ≤ income*4 per DATA-MODEL §6
        max_plan = self.income_cents * 4
        for code, cents in self.category_plans.items():
            if cents > max_plan:
                raise ValueError(
                    f"category_plans[{code!r}]={cents} exceeds income*4={max_plan}"
                )
        # Exactly one or zero accounts may be marked primary (zero → first auto-primary)
        primary_count = sum(1 for a in self.accounts if a.primary)
        if primary_count > 1:
            raise ValueError("At most one account may have primary=true")
        return self


class OnboardingV10Response(BaseModel):
    model_config = ConfigDict(extra="ignore")
    user_id: int
    income_cents: int
    account_ids: list[int]
    category_ids_by_code: dict[str, int]
    savings_category_id: int
    goal_id: Optional[int]
    savings_config: SavingsConfigReadInline
    onboarded_at: str


class SavingsConfigReadInline(BaseModel):
    """Inlined to avoid circular import with savings module."""
    roundup_enabled: bool
    roundup_base: int
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HTTP body → Pydantic | strict=True rejects implicit type coercion (e.g. "100" str→int); extra="forbid" rejects unknown fields |
| Pydantic → service | Service layer trusts validated types; defense-in-depth for known threats |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-22-12-01 | Tampering | Type confusion (string in int field) | mitigate | strict=True forbids implicit coercion; Pydantic v2 raises 422 |
| T-22-12-02 | Tampering | Unknown fields injecting state | mitigate | extra="forbid" rejects extra keys |
| T-22-12-03 | Tampering | Negative income_cents | mitigate | gt=0 constraint; surfaces 422 before service layer |
| T-22-12-04 | Tampering | category_plans with arbitrary codes | mitigate | @field_validator whitelist via VALID_DEFAULT_CODES |
| T-22-12-05 | Tampering | Σ plan > income via API | mitigate | @model_validator computes sum and rejects |
| T-22-12-06 | Tampering | Multiple primary accounts | mitigate | @model_validator counts and rejects > 1 |
| T-22-12-07 | Tampering | Goal due in past | mitigate | @field_validator on due field |
| T-22-12-08 | Information Disclosure | Account.mask in logs | accept | mask is non-sensitive (last 4 digits); not full PAN |
