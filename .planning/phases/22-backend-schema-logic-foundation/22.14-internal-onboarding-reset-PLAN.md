```python
"""/api/v1/internal/onboarding/reset endpoint (BE-15 admin reset).

CONTEXT D-07 / §Area 4: admin-only via X-Internal-Token, wipes user's accounts,
goals, savings_config, sets income_cents=NULL, plan_cents=0 — for повторного
onboarding в dev / тестах.

Mounted under internal_router with verify_internal_token gate inherited.
"""
from typing import Annotated

from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.ext.asyncio import AsyncSession

from app.api.dependencies import get_db
from app.db.session import set_tenant_scope
from app.services.onboarding_v10 import reset_v10


internal_onboarding_router = APIRouter(
    prefix="/onboarding",
    tags=["internal-onboarding"],
)


@internal_onboarding_router.delete("/reset", status_code=status.HTTP_200_OK)
async def reset_onboarding(
    user_id: int,  # query param
    db: Annotated[AsyncSession, Depends(get_db)],
) -> dict:
    """DELETE /api/v1/internal/onboarding/reset?user_id=N.

    Returns: {"user_id": N, "deleted_account_ids": [...]}
    """
    if user_id <= 0:
        raise HTTPException(status_code=422, detail="user_id must be positive integer")
    await set_tenant_scope(db, user_id)
    result = await reset_v10(db, user_id=user_id)
    return {"user_id": user_id, **result}
```

```python
"""Integration tests for DELETE /api/v1/internal/onboarding/reset (BE-15)."""
import pytest

@pytest.mark.asyncio
async def test_reset_without_token_returns_403(client):
    resp = await client.delete("/api/v1/internal/onboarding/reset?user_id=1")
    assert resp.status_code == 403


@pytest.mark.asyncio
async def test_reset_with_wrong_token_returns_403(client):
    resp = await client.delete(
        "/api/v1/internal/onboarding/reset?user_id=1",
        headers={"X-Internal-Token": "wrong"},
    )
    assert resp.status_code == 403


@pytest.mark.asyncio
async def test_reset_clears_accounts_goals_savings_config(
    client, internal_token_headers, seeded_v10_user_with_accounts,
):
    """End-to-end: complete v10, then reset, then verify all wiped."""
    user_id = seeded_v10_user_with_accounts.id
    resp = await client.delete(
        f"/api/v1/internal/onboarding/reset?user_id={user_id}",
        headers=internal_token_headers,
    )
    assert resp.status_code == 200
    body = resp.json()
    assert body["user_id"] == user_id
    assert isinstance(body["deleted_account_ids"], list)
    assert len(body["deleted_account_ids"]) >= 1


@pytest.mark.asyncio
async def test_reset_resets_income_to_null(
    client, internal_token_headers, db_session, seeded_v10_user_with_accounts,
):
    user_id = seeded_v10_user_with_accounts.id
    await client.delete(
        f"/api/v1/internal/onboarding/reset?user_id={user_id}",
        headers=internal_token_headers,
    )
    # Re-fetch user
    from sqlalchemy import select
    from app.db.models import AppUser
    user = await db_session.scalar(select(AppUser).where(AppUser.id == user_id))
    assert user.income_cents is None
    assert user.onboarded_at is None


@pytest.mark.asyncio
async def test_reset_then_complete_succeeds(
    client, internal_token_headers, auth_headers, seeded_v10_user_with_accounts,
):
    """After reset, POST /onboarding/complete succeeds (no 409)."""
    user_id = seeded_v10_user_with_accounts.id
    await client.delete(
        f"/api/v1/internal/onboarding/reset?user_id={user_id}",
        headers=internal_token_headers,
    )
    resp = await client.post(
        "/api/v1/onboarding/complete",
        headers=auth_headers,
        json={
            "income_cents": 200000_00,
            "accounts": [{"bank": "Т-Банк", "kind": "card", "balance_cents": 0}],
            "category_plans": {"food": 50000_00, "cafe": 25000_00},
        },
    )
    assert resp.status_code == 200


@pytest.mark.asyncio
async def test_reset_with_invalid_user_id_returns_422(client, internal_token_headers):
    resp = await client.delete(
        "/api/v1/internal/onboarding/reset?user_id=0",
        headers=internal_token_headers,
    )
    assert resp.status_code == 422
```
