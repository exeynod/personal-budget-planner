@pytest.mark.asyncio
async def test_after_upgrade_account_table_exists(upgraded_db):
    result = await upgraded_db.execute(
        text(
            "SELECT column_name, data_type FROM information_schema.columns "
            "WHERE table_name = 'account' ORDER BY ordinal_position"
        )
    )
    cols = {r[0]: r[1] for r in result.fetchall()}
    expected = {"id", "user_id", "bank", "mask", "kind", "balance_cents", "primary", "created_at"}
    assert expected.issubset(cols.keys()), f"Missing columns: {expected - cols.keys()}"
