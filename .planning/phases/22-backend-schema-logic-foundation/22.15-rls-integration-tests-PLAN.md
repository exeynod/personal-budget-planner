async def test_account_has_tenant_isolation_policy(db_session_admin):

async def test_v10_tables_force_rls_enabled(db_session_admin):

                "FROM pg_class WHERE relname = :t"

        )
        rls, force_rls = result.first()
        assert rls is True, f"{tbl} ENABLE ROW LEVEL SECURITY missing"
