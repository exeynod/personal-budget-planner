Shared SubscriptionsStore design (@MainActor @Observable, in Domain/Subscriptions/):
  final class SubscriptionsStore {
    struct API { listSubs; listCategories; listAccounts; reschedule; post; unpost; delete; patch; static let live = ... (proxies SubscriptionsV10API/Categories/Accounts) }
    init(api: API = .live, loadsCategoriesAccounts: Bool)   // V10 passes false (subs-only); v06 passes true
    private(set) var subscriptions/categories/accounts/status; private(set) var submitting
    inFlight + reloadPending re-entrancy (adopt the v06 superset — it's strictly safer than V10's; behavior for V10 is unchanged because reloadPending only triggers under concurrent load which V10 already serialized)
    func load() async                                        // conditionally fetches cats/accounts; sort + reschedule
    func post/unpost/delete(_:) async -> Bool                // submitting-guard + reload + WR-06
    func patch(id:payload:) async -> Bool
    // mutation result is reported via thrown/Bool — presentation copy stays in the VM
  }
