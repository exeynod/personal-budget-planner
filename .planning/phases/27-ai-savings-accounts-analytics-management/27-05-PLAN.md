       ```typescript
       import type { ActualV10Read, CategoryV10 } from '../../api/v10';

       export type GroupMode = 'day' | 'week' | 'cat';

       const MONTHS_RU_SHORT = ['ЯНВ','ФЕВ','МАР','АПР','МАЙ','ИЮН',
         'ИЮЛ','АВГ','СЕН','ОКТ','НОЯ','ДЕК'] as const;

       export interface MonthOption {
         label: string;       // "МАР 26"
         year: number;
         month: number;       // 1..12
         period_start: string; // YYYY-MM-01
         period_end: string;   // YYYY-MM-DD last day
       }

       export function lastNMonths(now: Date, n: number): MonthOption[] {
         const out: MonthOption[] = [];
         for (let i = n - 1; i >= 0; i--) {
           const d = new Date(now.getFullYear(), now.getMonth() - i, 1);
           const y = d.getFullYear();
           const m = d.getMonth() + 1;
           const yy = String(y).slice(-2);
           const lastDay = new Date(y, m, 0).getDate();
           out.push({
             label: `${MONTHS_RU_SHORT[m-1]} ${yy}`,
             year: y, month: m,
             period_start: `${y}-${String(m).padStart(2,'0')}-01`,
             period_end: `${y}-${String(m).padStart(2,'0')}-${String(lastDay).padStart(2,'0')}`,
           });
         }
         return out;
       }

       export function groupActualsByDay(actuals: ActualV10Read[], periodStart: string, periodEnd: string) {
         const filtered = actuals.filter((t) => t.tx_date >= periodStart && t.tx_date <= periodEnd && t.kind === 'expense');
         const map = new Map<string, number>();
         for (const t of filtered) {
           map.set(t.tx_date, (map.get(t.tx_date) || 0) + Math.abs(t.amount_cents));
         }
         return Array.from(map.entries()).sort(([a], [b]) => a < b ? -1 : 1).map(([key, sumCents]) => ({ key, sumCents }));
       }

       export function groupActualsByWeek(actuals: ActualV10Read[], periodStart: string) {
         const filtered = actuals.filter((t) => t.kind === 'expense');
         const map = new Map<number, number>();
         for (const t of filtered) {
           const day = parseInt(t.tx_date.slice(8, 10), 10);
           const weekIdx = Math.ceil(day / 7);  // 1..5
           map.set(weekIdx, (map.get(weekIdx) || 0) + Math.abs(t.amount_cents));
         }
         return Array.from(map.entries()).sort(([a], [b]) => a - b).map(([weekIdx, sumCents]) => ({ weekIdx, sumCents }));
       }

       export function groupActualsByCategory(actuals: ActualV10Read[], categories: CategoryV10[]) {
         const filtered = actuals.filter((t) => t.kind === 'expense');
         const catMap = new Map(categories.map((c) => [c.id, c]));
         const sumMap = new Map<number, number>();
         for (const t of filtered) {
           if (t.category_id == null) continue;
           sumMap.set(t.category_id, (sumMap.get(t.category_id) || 0) + Math.abs(t.amount_cents));
         }
         return Array.from(sumMap.entries())
           .map(([id, sumCents]) => ({
             category_id: id,
             category_name: catMap.get(id)?.name ?? '?',
             plan_cents: catMap.get(id)?.plan_cents ?? 0,
             sumCents,
           }))
           .sort((a, b) => b.sumCents - a.sumCents);
       }

       export function computeKPISpent(currActuals: ActualV10Read[], prevActuals: ActualV10Read[]) {
         const sum = (xs: ActualV10Read[]) => xs.filter(t => t.kind === 'expense').reduce((s, t) => s + Math.abs(t.amount_cents), 0);
         const sumCents = sum(currActuals);
         const prev = sum(prevActuals);
         const deltaCents = sumCents - prev;
         const deltaPct = prev === 0 ? 0 : Math.round((deltaCents / prev) * 100);
         return { sumCents, deltaCents, deltaPct };
       }

       export function computeKPISaved(actuals: ActualV10Read[], categories: CategoryV10[]) {
         const facts = groupActualsByCategory(actuals, categories);
         const factById = new Map(facts.map(f => [f.category_id, f.sumCents]));
         let saved = 0;
         for (const c of categories) {
           if (c.code === 'savings' || c.paused) continue;
           const fact = factById.get(c.id) ?? 0;
           saved += Math.max(0, (c.plan_cents ?? 0) - fact);
         }
         return { sumCents: saved };
       }

       export function shouldHighlightRed(barSum: number, barPlan: number, threshold = 0.75): boolean {
         if (barPlan <= 0) return false;
         return barSum / barPlan >= threshold;
       }

       export function computePct(sum: number, plan: number): number {
         if (plan <= 0) return 0;
         return Math.max(0, Math.min(100, Math.round((sum / plan) * 100)));
       }
       ```
