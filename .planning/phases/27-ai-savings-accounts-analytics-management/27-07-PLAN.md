    1. **`AiV10ViewModel.swift`** (NEW, ~180 LOC):
       ```swift
       import Foundation
       import Observation

       @MainActor
       @Observable
       final class AiV10ViewModel {
         enum Status: Equatable { case idle, loading, ready, error(String) }
         struct Message: Identifiable, Equatable {
           let id: String; let role: Role; var text: String
           enum Role { case user, ai }
         }

         var status: Status = .idle
         var observation: String? = nil
         var observationGeneratedAt: Date? = nil
         var observationError: String? = nil
         var messages: [Message] = []
         var isStreaming: Bool = false
         var input: String = ""
         private var inFlight: Bool = false

         func loadObservation() async {
           if inFlight { return }
           inFlight = true
           defer { inFlight = false }
           status = .loading
           do {
             let dto = try await AIObservationAPI.fetch()
             observation = dto.text
             observationGeneratedAt = dto.generatedAt
             status = .ready
           } catch {
             observationError = "Не удалось загрузить наблюдение"
             status = .ready  // Still allow chat
           }
         }

         func send(_ text: String) async {
           let trimmed = text.trimmingCharacters(in: .whitespacesAndNewlines)
           guard !trimmed.isEmpty, !isStreaming else { return }
           let userMsg = Message(id: "u-\(Date().timeIntervalSince1970)", role: .user, text: trimmed)
           messages.append(userMsg)
           input = ""
           let aiMsgId = "a-\(Date().timeIntervalSince1970)"
           messages.append(Message(id: aiMsgId, role: .ai, text: ""))
           isStreaming = true
           // Wire to v0.6 AIChatAPI streamChat:
           //   for await event in AIChatAPI.streamChat(message: trimmed) {
           //     switch event { case .token(let t): append to messages[aiMsgId]; case .done: break; case .error(let e): ... }
           //   }
           // Actual API discovery: read ios/BudgetPlanner/Features/AI/AIChatView.swift
           // and find the AsyncSequence / closure-based streaming function.
           // If v0.6 used callback + AbortController-like handle, adapt.
           do {
             try await runStream(message: trimmed, aiMsgId: aiMsgId)
           } catch {
             // append error suffix to ai message
             updateAiMsg(aiMsgId) { $0 += " ⚠ Ошибка" }
           }
           isStreaming = false
         }

         private func runStream(message: String, aiMsgId: String) async throws {
           // Adapter to v0.6 AIChatAPI. If v0.6 exposes AsyncStream<ChatEvent>:
           //   for await event in AIChatAPI.streamChat(message) { ... }
           // If callback-based: bridge via AsyncStream { continuation in ... }.
           // Implementation discretion — verify exact symbol by reading AIChatView.swift.
           //
           // Fallback if v0.6 doesn't expose easily: use direct URLSession SSE
           // against /api/v1/ai/chat — but this is bigger; prefer reuse.

           // PLACEHOLDER — real wiring depends on v0.6 surface:
           // for try await event in v0.6 AIChatAPI iterator {
           //   switch event { case let .token(t): updateAiMsg(aiMsgId) { $0 += t } ... }
           // }
         }

         private func updateAiMsg(_ id: String, _ mutate: (inout String) -> Void) {
           guard let idx = messages.firstIndex(where: { $0.id == id }) else { return }
           var msg = messages[idx]
           mutate(&msg.text)
           messages[idx] = msg
         }
       }
       ```
