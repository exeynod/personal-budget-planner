    1. **`frontend/src/screensV10/Ai/AiMount.tsx`** (NEW, ~150 LOC):
       ```typescript
       import { useEffect, useRef, useState } from 'react';
       import { fetchObservation } from '../../api/v10';
       import { streamChat } from '../../api/ai';
       import { usePosterRouter } from '../common';
       import { AiView, type AiMessage } from './AiView';
       import { todayRu, DEFAULT_SUGGESTION_CHIPS } from './computeAi';

       export function AiMount() {
         const router = usePosterRouter();
         const [observation, setObservation] = useState<string | null>(null);
         const [observationLoading, setObservationLoading] = useState(true);
         const [observationError, setObservationError] = useState<string | null>(null);
         const [observationGeneratedAt, setObservationGeneratedAt] = useState<Date | null>(null);
         const [messages, setMessages] = useState<AiMessage[]>([]);
         const [input, setInput] = useState('');
         const [isStreaming, setStreaming] = useState(false);
         const abortRef = useRef<AbortController | null>(null);

         useEffect(() => {
           let cancelled = false;
           setObservationLoading(true);
           fetchObservation()
             .then((res) => {
               if (cancelled) return;
               setObservation(res.text);
               setObservationGeneratedAt(new Date(res.generated_at));
               setObservationLoading(false);
             })
             .catch((e) => {
               if (cancelled) return;
               setObservationError('Не удалось загрузить наблюдение');
               setObservationLoading(false);
               // observation stays null — user still sees chips
             });
           return () => { cancelled = true; abortRef.current?.abort(); };
         }, []);

         function handleSend(text: string) {
           if (!text.trim() || isStreaming) return;
           const userMsg: AiMessage = { role: 'user', text, id: `u-${Date.now()}` };
           const aiMsgId = `a-${Date.now()}`;
           setMessages((prev) => [...prev, userMsg]);
           setInput('');
           setStreaming(true);
           let aiBuf = '';
           // Append empty AI bubble that we'll fill as tokens arrive.
           setMessages((prev) => [...prev, { role: 'ai', text: '', id: aiMsgId }]);
           const controller = new AbortController();
           abortRef.current = controller;
           streamChat(text, (event) => {
             // SSE event shape (from v0.6 api/ai.ts):
             //   { type: 'token', token: string } | { type: 'done' } | { type: 'error', error: string }
             if (event.type === 'token') {
               aiBuf += event.token;
               setMessages((prev) =>
                 prev.map((m) => (m.id === aiMsgId ? { ...m, text: aiBuf } : m))
               );
             } else if (event.type === 'done') {
               setStreaming(false);
             } else if (event.type === 'error') {
               aiBuf = aiBuf || '⚠ Ошибка ответа';
               setMessages((prev) =>
                 prev.map((m) => (m.id === aiMsgId ? { ...m, text: aiBuf } : m))
               );
               setStreaming(false);
             }
           }, controller.signal).catch(() => {
             setStreaming(false);
           });
         }

         function handleChipTap(chip: string) {
           handleSend(chip);
         }

         return (
           <AiView
             observation={observation}
             observationGeneratedAt={observationGeneratedAt}
             observationLoading={observationLoading}
             observationError={observationError}
             suggestionChips={DEFAULT_SUGGESTION_CHIPS}
             messages={messages}
             isStreaming={isStreaming}
             input={input}
             onInputChange={setInput}
             onSend={handleSend}
             onChipTap={handleChipTap}
             canPop={router.canPop}
             onBack={() => router.pop()}
             todayLabel={todayRu(new Date())}
           />
         );
       ```
