       ```typescript
       import type { AccountResponse, ActualV10Read } from '../../api/v10';

       export function sumAccountsBalances(list: AccountResponse[]): number {
         return list.reduce((acc, a) => acc + a.balance_cents, 0);
       }

       export function countAccounts(list: AccountResponse[]): number {
         return list.length;
       }

       export function formatBankSubtitle(a: AccountResponse): string {
         if (a.kind === 'cash') return 'наличные';
         if (a.kind === 'savings') return 'накопит. счёт';
         // card
         return a.mask ? `карта ·· ${a.mask}` : 'карта';
       }

       export function filterByAccount(actuals: ActualV10Read[], accountId: number): ActualV10Read[] {
         return actuals.filter((t) => t.account_id === accountId);
       }

       export function sumPeriodOps(
         actuals: ActualV10Read[], periodStart: string, periodEnd: string
       ): { count: number; sumCents: number } {
         const inRange = actuals.filter((t) => t.tx_date >= periodStart && t.tx_date <= periodEnd);
         return {
           count: inRange.length,
           sumCents: inRange.reduce((s, t) => s + Math.abs(t.amount_cents), 0),
         };
       }

       export function isValidNewAccountDraft(d: {
         bank: string; kind: string; balance_cents: number;
       }): boolean {
         return d.bank.trim().length > 0
           && (d.kind === 'card' || d.kind === 'cash' || d.kind === 'savings')
           && d.balance_cents >= 0;
       }
       ```
