       ```swift
           let isArchived: Bool?
           let planCents: Int?
           let rollover: CategoryRollover?
           let paused: Bool?

           // Custom encoder skips nil keys so backend gets only mutated fields.
           func encode(to encoder: Encoder) throws {
               var c = encoder.container(keyedBy: CodingKeys.self)
               try c.encodeIfPresent(name, forKey: .name)
               try c.encodeIfPresent(sortOrder, forKey: .sortOrder)
               try c.encodeIfPresent(isArchived, forKey: .isArchived)
               try c.encodeIfPresent(planCents, forKey: .planCents)
               try c.encodeIfPresent(rollover?.rawValue, forKey: .rollover)
               try c.encodeIfPresent(paused, forKey: .paused)
               try c.encodeIfPresent(parentId, forKey: .parentId)
           }

           private enum CodingKeys: String, CodingKey {
               case name, sortOrder, isArchived, planCents, rollover, paused, parentId
       ```

       ```swift
       import Foundation

       enum CategoryDetailData {
           /// Returns rounded percent over plan when fact > plan, else 0.
           static func computeOverPercent(factCents: Int, planCents: Int) -> Int {
               guard planCents > 0 else { return 0 }
               guard factCents > planCents else { return 0 }
               let pct = Double(factCents - planCents) / Double(planCents) * 100.0
               return Int(pct.rounded())
           }

           /// Returns rounded percent of plan used when fact ≤ plan.
           static func computeUnderPercent(factCents: Int, planCents: Int) -> Int {
               guard planCents > 0 else { return 0 }
               let pct = Double(factCents) / Double(planCents) * 100.0
               return Int(pct.rounded())
           }

           struct BarSegments: Equatable {
               let fillRatio: Double  // 0..1
               let tickAt: Double?    // 0..1 — nil = no tick
           }

           static func computeBarSegments(factCents: Int, planCents: Int) -> BarSegments {
               if factCents <= 0 { return BarSegments(fillRatio: 0, tickAt: nil) }
               if planCents <= 0 { return BarSegments(fillRatio: 1, tickAt: 0) }
               if factCents <= planCents {
                   return BarSegments(fillRatio: Double(factCents) / Double(planCents), tickAt: nil)
               }
               return BarSegments(fillRatio: 1, tickAt: Double(planCents) / Double(factCents))
           }

           static func filterActualsForCategory(_ actuals: [ActualV10DTO], categoryId: Int) -> [ActualV10DTO] {
               actuals.filter { $0.categoryId == categoryId }
           }

           static func computeFactForCategory(_ actuals: [ActualV10DTO], categoryId: Int) -> Int {
               actuals
                   .filter { $0.categoryId == categoryId && $0.kind == .expense }
                   .reduce(0) { $0 + abs($1.amountCents) }
           }
       }
       ```

       ```swift
       import Foundation
       import Observation

       @MainActor
       @Observable
       final class CategoryDetailViewModel {
           enum Status: Equatable {
               case idle, loading, ready, error(String)
           }

           let categoryId: Int
           private(set) var category: CategoryV10DTO?
           private(set) var actuals: [ActualV10DTO] = []
           private(set) var status: Status = .idle

           @ObservationIgnored var calendar: Calendar = .current
           private var inFlight = false

           init(categoryId: Int) {
               self.categoryId = categoryId
           }

           func load() async {
               guard !inFlight else { return }
               inFlight = true
               defer { inFlight = false }

               status = .loading
               do {
                   async let cats = CategoriesV10API.list()
                   async let period = PeriodsAPI.current()  // throws on 404 (mid-onboarding)

                   let allCats = try await cats
                   guard let cat = allCats.first(where: { $0.id == categoryId }) else {
                       status = .error("Категория не найдена")
                       return
                   }
                   self.category = cat

                   var per: PeriodDTO? = nil
                   do { per = try await period } catch { per = nil }
                   if let per {
                       let acts = try await ActualV10API.list(periodId: per.id)
                       self.actuals = acts
                   } else {
                       self.actuals = []
                   }
                   status = .ready
               } catch {
                   status = .error(error.localizedDescription)
               }
           }

           func toggleRollover() async {
               guard let cat = category else { return }
               let next: CategoryRollover = cat.rollover == .misc ? .savings : .misc
               do {
                   let updated = try await CategoriesV10API.update(
                       id: cat.id, payload: CategoryV10UpdateRequest(rollover: next)
                   )
                   self.category = updated
               } catch {
                   // For Plan 28 polish — toast/alert
               }
           }

           func togglePause() async {
               guard let cat = category else { return }
               do {
                   let updated = try await CategoriesV10API.update(
                       id: cat.id, payload: CategoryV10UpdateRequest(paused: !cat.paused)
                   )
                   self.category = updated
               } catch {
                   // For Plan 28 polish
               }
           }
       }
       ```
