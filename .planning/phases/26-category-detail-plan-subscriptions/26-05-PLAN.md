       ```swift
       import Foundation

       enum PlanData {
           static func computeSurplus(incomeCents: Int, plans: [PlanMonthItem]) -> Int {
               incomeCents - plans.reduce(0) { $0 + $1.planCents }
           }
           static func computeIsOverflow(_ surplus: Int) -> Bool { surplus < 0 }

           struct RolloverAggregates: Equatable {
               let miscCents: Int
               let savingsCents: Int
           }

           static func computeRolloverAggregates(
               categories: [CategoryV10DTO],
               plans: [PlanMonthItem],
               actuals: [ActualV10DTO]
           ) -> RolloverAggregates {
               let planByCat = Dictionary(uniqueKeysWithValues: plans.map { ($0.categoryId, $0.planCents) })
               var factByCat: [Int: Int] = [:]
               for a in actuals where a.kind == .expense {
                   factByCat[a.categoryId, default: 0] += abs(a.amountCents)
               }
               var misc = 0, sav = 0
               for c in categories where c.code != "savings" && !c.paused {
                   let plan = planByCat[c.id] ?? c.planCents
                   let fact = factByCat[c.id] ?? 0
                   let remainder = max(0, plan - fact)
                   if remainder == 0 { continue }
                   if c.rollover == .savings { sav += remainder } else { misc += remainder }
               }
               return RolloverAggregates(miscCents: misc, savingsCents: sav)
           }

           struct RegularRow: Identifiable, Equatable {
               let id: Int
               let name: String
               let dayOfMonth: Int
               let categoryName: String
               let amountCents: Int
               let postedTxnId: Int?
           }

           static func computeRegularsList(
               subs: [SubscriptionV10DTO],
               categories: [CategoryV10DTO]
           ) -> [RegularRow] {
               let nameById = Dictionary(uniqueKeysWithValues: categories.map { ($0.id, $0.name) })
               return subs
                   .filter { $0.cycle == .monthly && $0.dayOfMonth != nil }
                   .map { RegularRow(
                       id: $0.id, name: $0.name,
                       dayOfMonth: $0.dayOfMonth!,
                       categoryName: nameById[$0.categoryId] ?? "—",
                       amountCents: $0.amountCents,
                       postedTxnId: $0.postedTxnId
                   ) }
                   .sorted { $0.dayOfMonth < $1.dayOfMonth }
           }

           static func applyPlanEdit(_ plans: [PlanMonthItem], categoryId: Int, newCents: Int) -> [PlanMonthItem] {
               if let idx = plans.firstIndex(where: { $0.categoryId == categoryId }) {
                   var copy = plans
                   copy[idx] = PlanMonthItem(categoryId: categoryId, planCents: newCents)
                   return copy
               }
               return plans + [PlanMonthItem(categoryId: categoryId, planCents: newCents)]
           }

           static func plansFromCategories(_ cats: [CategoryV10DTO]) -> [PlanMonthItem] {
               cats
                   .filter { $0.code != "savings" && !$0.paused }
                   .map { PlanMonthItem(categoryId: $0.id, planCents: $0.planCents) }
           }
       }
       ```

       import Foundation
       import Observation

       @MainActor
       @Observable
       final class PlanViewModel {
           enum Status: Equatable {
               case idle, loading, ready, error(String)
           }

           let focusCategoryId: Int?

           private(set) var income: Int = 0
           private(set) var categories: [CategoryV10DTO] = []
           private(set) var actuals: [ActualV10DTO] = []
           private(set) var subs: [SubscriptionV10DTO] = []
           var plans: [PlanMonthItem] = []
           private(set) var status: Status = .idle
           var submitting: Bool = false
           var saveError: String? = nil
           var toastMessage: String? = nil

           @ObservationIgnored var calendar: Calendar = .current
           private var inFlight = false

           init(focusCategoryId: Int? = nil) {
               self.focusCategoryId = focusCategoryId
           }

           func load() async {
               guard !inFlight else { return }
               inFlight = true; defer { inFlight = false }
               status = .loading
               do {
                   async let cats = CategoriesV10API.list()
                   async let period = PeriodsAPI.current()
                   async let subsList = SubscriptionsV10API.list()
                   async let me = MeAPI.shared.get()  // see note re exact API name

                   let allCats = try await cats.filter { $0.code != "savings" && !$0.paused }
                                                  .sorted { ($0.ord ?? "99") < ($1.ord ?? "99") }
                   let s = try await subsList
                   let m = try await me

                   self.categories = allCats
                   self.subs = s
                   self.income = m.incomeCents ?? 0
                   self.plans = PlanData.plansFromCategories(allCats)

                   var per: PeriodDTO? = nil
                   do { per = try await period } catch { per = nil }
                   if let per {
                       self.actuals = try await ActualV10API.list(periodId: per.id)
                   }
                   status = .ready
               } catch {
                   status = .error(error.localizedDescription)
               }
           }

           func updateSlider(categoryId: Int, cents: Int) {
               plans = PlanData.applyPlanEdit(plans, categoryId: categoryId, newCents: cents)
           }

           func toggleRollover(categoryId: Int, to next: CategoryRollover) async {
               do {
                   let updated = try await CategoriesV10API.update(
                       id: categoryId, payload: CategoryV10UpdateRequest(rollover: next)
                   )
                   if let idx = categories.firstIndex(where: { $0.id == categoryId }) {
                       categories[idx] = updated
                   }
               } catch { /* polish: toast */ }
           }

           func postRegular(_ subId: Int) async {
               do {
                   _ = try await SubscriptionsV10API.post(id: subId)
                   toastMessage = "✓ ПРОВЕДЕНО → реестр"
                   await load()
               } catch { /* polish */ }
           }

           func unpostRegular(_ subId: Int) async {
               do {
                   try await SubscriptionsV10API.unpost(id: subId)
                   toastMessage = "Отменено"
                   await load()
               } catch { /* polish */ }
           }

           func submit() async -> Bool {
               submitting = true; defer { submitting = false }
               saveError = nil
               do {
                   _ = try await PlanMonthAPI.patch(plans: plans)
                   toastMessage = "✓ ПЛАН СОХРАНЁН"
                   return true
               } catch let APIError.http(status: 400, _) {  // adapt to actual APIError shape
                   saveError = "Σplan превышает доход — уменьшите лимиты"
                   return false
               } catch {
                   saveError = error.localizedDescription
                   return false
               }
           }
       }
