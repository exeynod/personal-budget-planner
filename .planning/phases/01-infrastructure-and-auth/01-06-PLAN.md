Создать `Dockerfile` в корне проекта:
```dockerfile
# Single Dockerfile for api/bot/worker containers (D-03)
# Build: docker build --build-arg SERVICE=api -t budget-api .
FROM python:3.12-slim AS base

ARG SERVICE
ENV SERVICE=${SERVICE}

# Install uv (fast Python package manager, D-01)
COPY --from=ghcr.io/astral-sh/uv:latest /uv /uvx /bin/

WORKDIR /app

# Copy dependency files first for layer caching
COPY pyproject.toml uv.lock* ./

# Install Python dependencies (all services share same deps)
RUN uv sync --locked --no-install-project --no-dev 2>/dev/null || uv sync --no-install-project --no-dev

# Copy application code
COPY app/ ./app/
COPY alembic/ ./alembic/
COPY alembic.ini ./

COPY main_api.py main_bot.py main_worker.py ./
COPY entrypoint.sh ./
RUN chmod +x entrypoint.sh

# Default CMD selects service based on SERVICE build-arg
# Using shell form to support variable expansion
CMD if [ "$SERVICE" = "api" ]; then \
      ./entrypoint.sh; \
    elif [ "$SERVICE" = "bot" ]; then \
      uv run python main_bot.py; \
    elif [ "$SERVICE" = "worker" ]; then \
      uv run python main_worker.py; \
    else \
      echo "Unknown SERVICE: $SERVICE (must be api/bot/worker)" && exit 1; \
    fi
```

```
# Production Caddyfile — TLS via Let's Encrypt (INF-03)
# PUBLIC_DOMAIN is injected via docker-compose environment: PUBLIC_DOMAIN=${PUBLIC_DOMAIN}
{env.PUBLIC_DOMAIN} {

    # Reverse proxy all /api/* to FastAPI (api container)
    # NOTE: /api/v1/internal/* is accessible within docker network only;
    # Caddy proxies all /api/* but internal/* is protected by X-Internal-Token dependency
    reverse_proxy /api/* api:8000

    # SPA fallback: serve React frontend static files from /srv/dist
    root * /srv/dist
    try_files {path} /index.html
    file_server
}
```
