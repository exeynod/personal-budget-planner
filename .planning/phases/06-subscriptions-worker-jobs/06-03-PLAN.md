  <action>
    Создать `app/api/routes/subscriptions.py` per D-71:

    ```python
    from fastapi import APIRouter, Depends, HTTPException, status
    from app.api.dependencies import get_current_user, get_db
    from app.api.schemas.subscriptions import SubscriptionCreate, SubscriptionUpdate, SubscriptionRead, ChargeNowResponse
    from app.services import subscriptions as sub_service
    from app.services.settings import get_cycle_start_day

    router = APIRouter(prefix="/subscriptions", tags=["subscriptions"], dependencies=[Depends(get_current_user)])

    @router.get("", response_model=list[SubscriptionRead])
    async def list_subs(db=Depends(get_db)):
        return await sub_service.list_subscriptions(db)

    @router.post("", response_model=SubscriptionRead)
    async def create_sub(payload: SubscriptionCreate, db=Depends(get_db), user=Depends(get_current_user)):
        try:
            sub = await sub_service.create_subscription(
                db, tg_user_id=user.tg_user_id,
                name=payload.name, amount_cents=payload.amount_cents,
                cycle=payload.cycle, next_charge_date=payload.next_charge_date,
                category_id=payload.category_id, notify_days_before=payload.notify_days_before,
                is_active=payload.is_active,
            )
            await db.commit()
            return sub
        except sub_service.CategoryNotFoundOrArchived:
            raise HTTPException(400, "category not found or archived")

    @router.patch("/{sub_id}", response_model=SubscriptionRead)
    async def patch_sub(sub_id: int, payload: SubscriptionUpdate, db=Depends(get_db)):
        try:
            sub = await sub_service.update_subscription(db, sub_id, payload.model_dump(exclude_unset=True))
            await db.commit()
            return sub
        except LookupError:
            raise HTTPException(404, "subscription not found")

    @router.delete("/{sub_id}", status_code=204)
    async def delete_sub(sub_id: int, db=Depends(get_db)):
        await sub_service.delete_subscription(db, sub_id)
        await db.commit()

    @router.post("/{sub_id}/charge-now", response_model=ChargeNowResponse)
    async def charge_now(sub_id: int, db=Depends(get_db), user=Depends(get_current_user)):
        cycle_start = await get_cycle_start_day(db, user.tg_user_id)
        try:
            planned, new_date = await sub_service.charge_subscription(db, sub_id, cycle_start_day=cycle_start)
            await db.commit()
            return ChargeNowResponse(planned_id=planned.id, next_charge_date=new_date)
        except sub_service.AlreadyChargedError:
            raise HTTPException(status.HTTP_409_CONFLICT, "already charged for this date")
        except LookupError:
            raise HTTPException(404, "subscription not found")
    ```

    Использовать паттерн `await db.commit()` после успешной операции (как в planned.py / actual.py).
