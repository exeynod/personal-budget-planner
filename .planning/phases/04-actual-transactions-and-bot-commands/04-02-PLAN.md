    tg_user_id: int
    kind: KindStr
    amount_cents: int = Field(gt=0)
    description: Optional[str] = Field(default=None, max_length=500)
    tx_date: Optional[date] = None
    category_query: Optional[str] = Field(default=None, max_length=200)
    category_id: Optional[int] = Field(default=None, gt=0)

    @model_validator(mode="after")
    def _check_either_query_or_id(self) -> "BotActualRequest":
        if not self.category_query and not self.category_id:
            raise ValueError("Either category_query or category_id required")
        return self


class CategoryCandidate(BaseModel):
    id: int
    name: str
    kind: KindStr


class BotActualResponse(BaseModel):
    """Discriminated by `status`. created → actual+category populated; ambiguous → candidates."""
    status: Literal["created", "ambiguous", "not_found"]
    actual: Optional[ActualRead] = None
    category: Optional[CategoryCandidate] = None
    category_balance_cents: Optional[int] = None  # remaining_cents (planned - sum_actual для expense; sum_actual - planned для income — D-02 sign rule)
    candidates: Optional[list[CategoryCandidate]] = None

    tg_user_id: int

    period_id: int

    actual_total_expense_cents: int
    planned_total_income_cents: int
    actual_total_income_cents: int
    by_category: list[BalanceCategoryRow]


class BotTodayActualRow(BaseModel):
    id: int
    kind: KindStr
    amount_cents: int
    description: Optional[str]

async def process_bot_actual(
    db,
    *,
    tg_user_id: int,
    kind: str,
    amount_cents: int,
    category_query: Optional[str] = None,
    category_id: Optional[int] = None,
    description: Optional[str] = None,
    tx_date: Optional[date] = None,
) -> dict:
    """Returns dict matching BotActualResponse shape.

    Flow:
    1. If category_id provided:
       - cat = await categories.get_or_404(db, category_id) → если archived raise InvalidCategoryError
       - candidates = [cat]
    2. Else (category_query):
       - candidates = await actual.find_categories_by_query(db, category_query)
    3. Branch:
       - 0 candidates → return {"status": "not_found", "candidates": []}
       - >1 candidates → return {"status": "ambiguous", "candidates": [{id, name, kind} for c in candidates]}
       - 1 candidate → продолжаем:
           - tx_date = tx_date or _today_in_app_tz()
           - actual_row = await actual.create_actual(db, kind=kind, amount_cents=amount_cents,
                description=description, category_id=cat.id, tx_date=tx_date, source=ActualSource.bot)
           - balance_after = compute_category_balance(db, period_id=actual_row.period_id, category_id=cat.id, kind=kind)
                # Helper: SUM(planned where category=X AND kind=Y) - SUM(actual where category=X AND kind=Y) для expense (отрицательные = overspend);
                # для income — обратное (act - plan).
           - return {"status": "created", "actual": ActualRead.model_validate(actual_row), "category": {...}, "category_balance_cents": balance_after}
    """

async def format_balance_for_bot(db, *, tg_user_id: int) -> dict:
    """Returns dict matching BotBalanceResponse. Wraps actual.compute_balance for active period."""
    period = await periods.get_current_active_period(db)
    if period is None:
        raise PeriodNotFoundError(0)  # bot route maps to 404
    bal = await actual.compute_balance(db, period.id)
    # Strip BalanceResponse-specific fields (`starting_balance_cents`) if not needed; OR just pass through;
    # spec: BotBalanceResponse omits starting_balance_cents field intentionally — keep only what bot needs.
    return {
        "period_id": bal["period_id"],
        "period_start": bal["period_start"],
        "period_end": bal["period_end"],
        "balance_now_cents": bal["balance_now_cents"],
        "delta_total_cents": bal["delta_total_cents"],
        "planned_total_expense_cents": bal["planned_total_expense_cents"],
        "actual_total_expense_cents": bal["actual_total_expense_cents"],
        "planned_total_income_cents": bal["planned_total_income_cents"],
        "actual_total_income_cents": bal["actual_total_income_cents"],
        "by_category": bal["by_category"],
    }

async def format_today_for_bot(db, *, tg_user_id: int) -> dict:
    """Returns dict matching BotTodayResponse. Wraps actual.actuals_for_today + nested category names."""
    rows = await actual.actuals_for_today(db)
    # Need category name; bulk-fetch via SELECT IN (category_ids) or eager-load through relationship.
    cat_ids = list({r.category_id for r in rows})
    cats = {c.id: c for c in (await db.execute(select(Category).where(Category.id.in_(cat_ids)))).scalars().all()}
    actuals_out = []
    for r in rows:
        cat = cats.get(r.category_id)
        actuals_out.append({
            "id": r.id,
            "kind": r.kind.value,
            "amount_cents": r.amount_cents,
            "description": r.description,
            "category_id": r.category_id,
            "category_name": cat.name if cat else "(archived)",
        })
    total_expense = sum(r.amount_cents for r in rows if r.kind == CategoryKind.expense)
    total_income = sum(r.amount_cents for r in rows if r.kind == CategoryKind.income)
    return {"actuals": actuals_out, "total_expense_cents": total_expense, "total_income_cents": total_income}
