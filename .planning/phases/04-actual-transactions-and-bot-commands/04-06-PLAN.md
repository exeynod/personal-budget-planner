```typescript
function groupByDate(rows: ActualRead[]): {date: string; label: string; rows: ActualRead[]}[] {
  const todayISO = new Date().toISOString().slice(0, 10);
  const y = new Date(); y.setDate(y.getDate() - 1);
  const yesterdayISO = y.toISOString().slice(0, 10);

  const map = new Map<string, ActualRead[]>();
  for (const r of rows) {
    const arr = map.get(r.tx_date) ?? [];
    arr.push(r);
    map.set(r.tx_date, arr);
  }
  const sorted = [...map.entries()].sort((a, b) => b[0].localeCompare(a[0]));
  return sorted.map(([date, rs]) => ({
    date,
    label: date === todayISO ? 'Сегодня' :
           date === yesterdayISO ? 'Вчера' :
           new Date(date).toLocaleDateString('ru-RU', { day: 'numeric', month: 'long' }),
    rows: rs.sort((a, b) => b.id - a.id),  // newest first within day
  }));
}
```

```typescript
export interface ActualScreenProps {
  onBack: () => void;
}

interface SheetState {
  open: boolean;
  mode: 'create' | 'edit';
  item?: ActualRead;
}
const CLOSED_SHEET: SheetState = { open: false, mode: 'create' };

function groupByDate(rows: ActualRead[]): {date: string; label: string; rows: ActualRead[]}[] {
  // см. <interfaces>
}

function formatRubles(cents: number): string {
  return (cents / 100).toLocaleString('ru-RU', { maximumFractionDigits: 2 });
}

export function ActualScreen({ onBack }: ActualScreenProps) {
  const { period, loading: perLoading, error: perError } = useCurrentPeriod();
  const { rows, loading: rowsLoading, error: rowsError, refetch } = useActual(period?.id ?? null);
  const { categories } = useCategories(false);

  const [sheet, setSheet] = useState<SheetState>(CLOSED_SHEET);
  const [toast, setToast] = useState<string | null>(null);
  const [busy, setBusy] = useState(false);
  const [mutationError, setMutationError] = useState<string | null>(null);

  const groups = useMemo(() => groupByDate(rows), [rows]);

  const showToast = (msg: string) => {
    setToast(msg);
    window.setTimeout(() => setToast(null), 2200);
  };

  const openCreate = () => setSheet({ open: true, mode: 'create', item: undefined });
  const openEdit = (item: ActualRead) => setSheet({ open: true, mode: 'edit', item });

  const handleSave = async (data: ActualEditorSavePayload) => {
    setMutationError(null);
    if (sheet.mode === 'create') {
      await createActual(data);
    } else if (sheet.item) {
      await updateActual(sheet.item.id, data);
    }
    setSheet(CLOSED_SHEET);
    await refetch();
    showToast(sheet.mode === 'create' ? 'Записано' : 'Сохранено');
  };

  const handleDelete = async () => {
    if (!sheet.item) return;
    setMutationError(null);
    await deleteActual(sheet.item.id);
    setSheet(CLOSED_SHEET);
    await refetch();
    showToast('Удалено');
  };

  // Period label
  const periodLabel = period
    ? `${new Date(period.period_start).toLocaleDateString('ru-RU', { month: 'long', year: 'numeric' })} · `
      + `${new Date(period.period_start).toLocaleDateString('ru-RU', { day: '2-digit', month: 'short' })} — `
      + `${new Date(period.period_end).toLocaleDateString('ru-RU', { day: '2-digit', month: 'short' })}`
    : '';

  // Render branches
  if (perLoading) return <div className={styles.muted}>Загрузка периода…</div>;
  if (perError) return <div className={styles.error}>Ошибка периода: {perError}</div>;
  if (!period) {
    return (
      <div className={styles.root}>
        <header className={styles.header}>
          <button type="button" onClick={onBack} className={styles.backBtn} aria-label="Назад">←</button>
          <div className={styles.title}>Факт</div>
        </header>
        <div className={styles.empty}>Сначала завершите onboarding.</div>
      </div>
    );
  }

  const isEmpty = !rowsLoading && rows.length === 0;

  return (
    <div className={styles.root}>
      <header className={styles.header}>
        <button type="button" onClick={onBack} className={styles.backBtn} aria-label="Назад">←</button>
        <div className={styles.titleBlock}>
          <div className={styles.title}>Факт периода</div>
          <div className={styles.subtitle}>{periodLabel}</div>
        </div>
      </header>

      {rowsError && <div className={styles.error}>Ошибка списка: {rowsError}</div>}
      {mutationError && <div className={styles.error}>Ошибка: {mutationError}</div>}

      {isEmpty && (
        <div className={styles.empty}>Пока нет факт-трат. Нажмите ＋ чтобы добавить.</div>
      )}

      {groups.map((g) => (
        <section key={g.date} className={styles.dateGroup}>
          <h3 className={styles.dateLabel}>{g.label}</h3>
          {g.rows.map((row) => {
            const cat = categories.find((c) => c.id === row.category_id);
            const catName = cat?.name ?? '(категория удалена)';
            return (
              <button
                key={row.id}
                type="button"
                className={`${styles.row} ${row.kind === 'income' ? styles.incomeRow : ''}`}
                onClick={() => openEdit(row)}
              >
                <span className={styles.amount}>{formatRubles(row.amount_cents)} ₽</span>
                <span className={styles.category}>{catName}</span>
                {row.description && <span className={styles.desc}>{row.description}</span>}
              </button>
            );
          })}
        </section>
      ))}

      <Fab onClick={openCreate} ariaLabel="Добавить трату" />

      {toast && <div className={styles.toast}>{toast}</div>}

      <BottomSheet
        open={sheet.open}
        onClose={() => setSheet(CLOSED_SHEET)}
        title={sheet.mode === 'edit' ? 'Изменить трату' : 'Новая трата'}
      >
        <ActualEditor
          initial={
            sheet.item
              ? {
                  kind: sheet.item.kind,
                  amount_cents: sheet.item.amount_cents,
                  description: sheet.item.description,
                  category_id: sheet.item.category_id,
                  tx_date: sheet.item.tx_date,
                }
              : undefined
          }
          categories={categories}
          onSave={handleSave}
          onDelete={sheet.mode === 'edit' ? handleDelete : undefined}
          onCancel={() => setSheet(CLOSED_SHEET)}
        />
      </BottomSheet>
    </div>
  );
}
```
