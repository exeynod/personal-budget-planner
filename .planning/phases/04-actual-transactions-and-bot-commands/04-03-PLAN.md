### app/api/routes/actual.py — endpoints

```python
actual_router = APIRouter(
    tags=["actual"],
    dependencies=[Depends(get_current_user)],
)

@actual_router.get("/periods/{period_id}/actual", response_model=list[ActualRead])
async def list_actual(
    period_id: int,
    db: Annotated[AsyncSession, Depends(get_db)],
    kind: Optional[KindStr] = Query(default=None),
    category_id: Optional[int] = Query(default=None, gt=0),
) -> list[ActualRead]:
    rows = await actual_svc.list_actual_for_period(db, period_id, kind=kind, category_id=category_id)
    return [ActualRead.model_validate(r) for r in rows]

@actual_router.post("/actual", response_model=ActualRead, status_code=200)
async def create_actual(
    body: ActualCreate,
    db: Annotated[AsyncSession, Depends(get_db)],
) -> ActualRead:
    """source=mini_app forced server-side (D-53)."""
    try:
        row = await actual_svc.create_actual(
            db,
            kind=body.kind, amount_cents=body.amount_cents, description=body.description,
            category_id=body.category_id, tx_date=body.tx_date,
            source=ActualSource.mini_app,
        )
    except CategoryNotFoundError as exc: raise HTTPException(404, str(exc)) from exc
    except InvalidCategoryError as exc: raise HTTPException(400, str(exc)) from exc
    except KindMismatchError as exc: raise HTTPException(400, str(exc)) from exc
    except FutureDateError as exc: raise HTTPException(400, str(exc)) from exc
    return ActualRead.model_validate(row)

@actual_router.patch("/actual/{actual_id}", response_model=ActualRead)
async def update_actual(
    actual_id: int,
    body: ActualUpdate,
    db: Annotated[AsyncSession, Depends(get_db)],
) -> ActualRead:
    try:
        row = await actual_svc.update_actual(db, actual_id, body)
    except ActualNotFoundError as exc: raise HTTPException(404, str(exc)) from exc
    except CategoryNotFoundError as exc: raise HTTPException(404, str(exc)) from exc
    except InvalidCategoryError as exc: raise HTTPException(400, str(exc)) from exc
    except KindMismatchError as exc: raise HTTPException(400, str(exc)) from exc
    except FutureDateError as exc: raise HTTPException(400, str(exc)) from exc
    return ActualRead.model_validate(row)

@actual_router.delete("/actual/{actual_id}", response_model=ActualRead)
async def delete_actual(
    actual_id: int,
    db: Annotated[AsyncSession, Depends(get_db)],
) -> ActualRead:
    try:
        row = await actual_svc.delete_actual(db, actual_id)
    except ActualNotFoundError as exc: raise HTTPException(404, str(exc)) from exc
    return ActualRead.model_validate(row)

@actual_router.get("/actual/balance", response_model=BalanceResponse)
async def get_balance(
    db: Annotated[AsyncSession, Depends(get_db)],
) -> BalanceResponse:
    """Returns balance for the active period; 404 if no active period exists."""
    period = await periods_svc.get_current_active_period(db)
    if period is None:
        raise HTTPException(404, "No active budget period — complete onboarding first")
    bal = await actual_svc.compute_balance(db, period.id)
    return BalanceResponse(**bal)
```

**Note on URL ordering:** FastAPI matches first-defined-first. Поэтому `/actual/balance` ОБЯЗАТЕЛЬНО ДО `/actual/{actual_id}` (иначе `/actual/balance` будет matchen как `actual_id="balance"` → 422). Order matters.

internal_bot_router = APIRouter(prefix="/bot", tags=["internal-bot"])

    except KindMismatchError as exc: raise HTTPException(400, str(exc)) from exc
    except FutureDateError as exc: raise HTTPException(400, str(exc)) from exc
    return BotActualResponse(**result)

@internal_bot_router.post("/balance", response_model=BotBalanceResponse)
async def bot_balance(
    body: BotBalanceRequest,

) -> BotBalanceResponse:

        result = await internal_bot_svc.format_balance_for_bot(db, tg_user_id=body.tg_user_id)
    except PeriodNotFoundError as exc:

    return BotBalanceResponse(**result)

@internal_bot_router.post("/today", response_model=BotTodayResponse)

    body: BotTodayRequest,
