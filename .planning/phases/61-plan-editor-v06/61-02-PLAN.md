    enum PlanEditorData {

        // MARK: - Surplus

        /// incomeCents − Σ(planCents) over non-paused expense categories.
        /// Income kind categories НЕ вычитаются (они приносят дополнительный
        /// план-доход поверх monthly income — separate flow).
        /// Archived и paused — исключены.
        static func computeSurplus(
            incomeCents: Int,
            categories: [CategoryV10DTO]
        ) -> Int {
            let sumExpensePlan = categories
                .filter { !$0.isArchived && !$0.paused && $0.kind == .expense }
                .reduce(0) { $0 + $1.planCents }
            return incomeCents - sumExpensePlan
        }

        // MARK: - Display sorting

        /// Returns (expense, income) tuple. Internal sort:
        ///   1. paused == false first, paused == true at end (within kind)
        ///   2. ord ?? "99" ASC
        ///   3. tie-break by name ASC
        /// Archived excluded.
        static func sortCategoriesForDisplay(
            _ categories: [CategoryV10DTO]
        ) -> (expense: [CategoryV10DTO], income: [CategoryV10DTO]) {
            let active = categories.filter { !$0.isArchived }
            let sorted = active.sorted { a, b in
                if a.paused != b.paused { return !a.paused }  // false < true
                let oa = a.ord ?? "99"
                let ob = b.ord ?? "99"
                if oa != ob { return oa < ob }
                return a.name < b.name
            }
            let expense = sorted.filter { $0.kind == .expense }
            let income = sorted.filter { $0.kind == .income }
            return (expense: expense, income: income)
        }

        // MARK: - Fact lookup

        /// Σ abs(amountCents) по actuals где actual.categoryId == categoryId
        /// (любой kind — .expense / .income / .roundup / .deposit).
        static func factCentsByCategory(
            _ actuals: [ActualV10DTO],
            categoryId: Int
        ) -> Int {
            actuals
                .filter { $0.categoryId == categoryId }
                .reduce(0) { $0 + abs($1.amountCents) }
        }

        // MARK: - Rollover aggregates

        struct RolloverAggregates: Equatable {
            let miscCents: Int
            let savingsCents: Int
        }

        /// Partition (plan − fact) leftover by category.rollover.
        /// Excludes:
        ///   - archived categories
        ///   - paused categories
        ///   - code == "savings" (системная корзина — separate flow)
        ///   - over-budget rows: remainder = max(0, plan − fact)
        /// Considers only .expense kind (income категории не have rollover semantics).
        static func computeRolloverAggregates(
            categories: [CategoryV10DTO],
            actuals: [ActualV10DTO]
        ) -> RolloverAggregates {
            var factByCat: [Int: Int] = [:]
            for a in actuals where a.kind == .expense {
                factByCat[a.categoryId, default: 0] += abs(a.amountCents)
            }
            var misc = 0
            var sav = 0
            for c in categories
                where !c.isArchived && !c.paused
                && c.kind == .expense
                && c.code != "savings"
            {
                let fact = factByCat[c.id] ?? 0
                let remainder = Swift.max(0, c.planCents - fact)
                if remainder == 0 { continue }
                switch c.rollover {
                case .savings: sav += remainder
                case .misc: misc += remainder
                }
            }
            return RolloverAggregates(miscCents: misc, savingsCents: sav)
        }

        // MARK: - Optimistic update

        /// Заменяет CategoryV10DTO в списке по id. Если id не найден —
        /// возвращает исходный массив без изменений (no append).
        /// Immutable: input array не мутируется.
        static func applyOptimisticUpdate(
            _ categories: [CategoryV10DTO],
            updated: CategoryV10DTO
        ) -> [CategoryV10DTO] {
            guard let idx = categories.firstIndex(where: { $0.id == updated.id }) else {
                return categories
            }
            var copy = categories
            copy[idx] = updated
            return copy
        }
    }
