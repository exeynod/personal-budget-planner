```python
async def get_recent_messages(
    db: AsyncSession,
    conv_id: int,
    *,
    user_id: int,
    limit: int = 20,
) -> list[AiMessage]:
    # AiMessage scoped по conversation, но добавляем user_id для defense
    q = (
        select(AiMessage)
        .where(
            AiMessage.conversation_id == conv_id,
            AiMessage.user_id == user_id,
        )
        .order_by(AiMessage.id.desc())
        .limit(limit)
    )
    rows = (await db.execute(q)).scalars().all()
    return list(reversed(rows))


async def clear_messages(db: AsyncSession, conv_id: int, *, user_id: int) -> int:
    """Hard delete всех ai_message для conversation. Conversation row остаётся."""
    result = await db.execute(
        delete(AiMessage).where(
            AiMessage.conversation_id == conv_id,
            AiMessage.user_id == user_id,
        )
    )
    return result.rowcount or 0
```

```python
async def upsert_category_embedding(
    db: AsyncSession, category_id: int, vector: list[float], *, user_id: int
) -> None:
    """INSERT or UPDATE category_embedding. user_id заполняется явно для NOT NULL constraint."""
    # Существующая логика INSERT ... ON CONFLICT (category_id) DO UPDATE
    # Нужно добавить user_id в INSERT clause.
    stmt = pg_insert(CategoryEmbedding).values(
        category_id=category_id,
        user_id=user_id,
        embedding=vector,
        updated_at=func.now(),
    ).on_conflict_do_update(
        index_elements=["category_id"],
        set_={"embedding": vector, "updated_at": func.now(), "user_id": user_id},
    )
    await db.execute(stmt)


async def find_similar_categories(
    db: AsyncSession, query_vector: list[float], *, user_id: int, limit: int = 5
) -> list[tuple[int, float]]:
    """Cosine-similarity search по embeddings, scoped по user_id."""
    stmt = (
        select(CategoryEmbedding.category_id, CategoryEmbedding.embedding.cosine_distance(query_vector).label("dist"))
        .where(CategoryEmbedding.user_id == user_id)
        .order_by("dist")
        .limit(limit)
    )
    rows = (await db.execute(stmt)).all()
    return [(r.category_id, 1.0 - r.dist) for r in rows]
```
