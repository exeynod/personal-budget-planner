# app/api/dependencies.py
async def get_current_user_id(
    current_user: Annotated[dict, Depends(get_current_user)],
    db: Annotated[AsyncSession, Depends(get_db)],
) -> int:
    """Resolve app_user.id (PK) by tg_user_id из current_user dict.

    Cached на request через FastAPI dep граф (FastAPI кэширует Depends результаты
    автоматически в рамках одного request).

    Raises HTTPException(403) если AppUser строка не найдена — Phase 12 заменит
    на role-check; в Phase 11 поведение совпадает с OWNER_TG_ID whitelist.
    """
    tg_user_id = current_user["id"]
    result = await db.execute(
        select(AppUser.id).where(AppUser.tg_user_id == tg_user_id)
    )
    user_id = result.scalar_one_or_none()
    if user_id is None:
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="AppUser not found for current Telegram user — onboarding required",
        )
    return user_id


async def get_db_with_tenant_scope(
    user_id: Annotated[int, Depends(get_current_user_id)],
) -> AsyncGenerator[AsyncSession, None]:
    """Yield AsyncSession + SET LOCAL app.current_user_id = :user_id.

    После SET LOCAL все queries в рамках этой transaction матчат RLS policy
    (user_id = current_setting('app.current_user_id')::bigint). На COMMIT/ROLLBACK
    setting автоматически сбрасывается (LOCAL = transaction scope).

    Этот dep заменяет get_db в роутах с user-scope. Public/internal-token
    эндпоинты продолжают использовать get_db.
    """
    async with AsyncSessionLocal() as session:
        try:
            await set_tenant_scope(session, user_id)
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise


# app/db/session.py
async def set_tenant_scope(session: AsyncSession, user_id: int) -> None:
    """SET LOCAL app.current_user_id = :user_id для текущей transaction."""
    await session.execute(
        text("SET LOCAL app.current_user_id = :uid"),
        {"uid": user_id},
    )
```

async def set_tenant_scope(session: AsyncSession, user_id: int) -> None:
    """Установить app.current_user_id GUC для текущей transaction (Phase 11 MUL-02).

    Должно быть вызвано до любого query, который затрагивает доменные таблицы
    (категории, периоды, транзакции, подписки, AI tables) — иначе RLS policy
    coalesce(current_setting('app.current_user_id', true)::bigint, -1) даст -1
    и query вернёт 0 строк.

    SET LOCAL — transaction scope: при COMMIT/ROLLBACK значение сбрасывается.
    Используется в:
      - app/api/dependencies.py::get_db_with_tenant_scope (per-request).
      - app/worker/jobs/* (per-tenant iteration через explicit set + commit per tenant).

    Args:
        session: AsyncSession в открытой transaction.
        user_id: app_user.id (PK), не tg_user_id.
    """
    from sqlalchemy import text

    await session.execute(
        text("SET LOCAL app.current_user_id = :uid"),
        {"uid": user_id},
    )
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| FastAPI request → DB session | get_db_with_tenant_scope ставит SET LOCAL ДО первого query — атакующий не может отправить SQL который выполнится без RLS scope |
| Worker → DB session | set_tenant_scope helper used by worker; worker сам выбирает корректный user_id (validated, не из user input) |
| Dev seed → DB | dev_seed.py запускается только при DEV_MODE=true (caller проверяет); production не запускает |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-04-01 | Spoofing | Атакующий отправляет произвольный user_id в SET LOCAL bypass'ив RLS | mitigate | get_current_user_id резолвит app_user.id из validated tg_user_id (HMAC-проверенный initData); user_id никогда не приходит из request body. set_tenant_scope использует bind parameter — не string concat. |
| T-11-04-02 | Tampering | Race: SET LOCAL применяется НЕ в той же transaction что и query | mitigate | get_db_with_tenant_scope открывает session, в той же блокировке делает set_tenant_scope, потом yield. AsyncSession без autocommit держит transaction открытой до commit/rollback. |
| T-11-04-03 | Information Disclosure | Connection pool reuse — SET LOCAL «утекает» в следующий request | mitigate | SET LOCAL — transaction scope; на await session.commit() / rollback() значение сбрасывается. Connection возвращается в пул чистым. Verified в Plan 11-07 test_rls_setting_resets_after_commit. |
| T-11-04-04 | Repudiation | get_current_user_id возвращает 403 без логирования | mitigate | HTTPException(403) — FastAPI логирует HTTP status автоматически через uvicorn; detail сообщает причину для debug; никаких sensitive id не утекает в response (только generic message). |
| T-11-04-05 | Elevation of Privilege | dev_seed запускается на production и подменяет role | accept | dev_seed.py — caller-controlled (app.main_api.lifespan проверяет settings.DEV_MODE перед вызовом). Production deploy не имеет DEV_MODE=true. CLAUDE.md memory feedback подтверждает: secrets/dev — пользователь сам управляет. |
| T-11-04-06 | DoS | get_current_user_id выполняет дополнительный SELECT на каждый request | accept | One SELECT(id) WHERE indexed_unique_column. Pet-проект, latency budget allows. Phase 12 может cache'ить через REQUEST scope если станет hot. |
