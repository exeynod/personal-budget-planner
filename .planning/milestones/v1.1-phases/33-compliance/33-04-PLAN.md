    ```python
    """Phase 33 CMP-33-06: data export builder.

    Constructs a JSON-serializable dict containing all PII data for one user.
    The shape is per CONTEXT.md `data export response`. RLS-scope must be set
    before calling — все SELECT's идут под `app.current_user_id` GUC, который
    автоматически filterает rows.
    """
    from __future__ import annotations

    from datetime import datetime
    from typing import Any

    from sqlalchemy import select
    from sqlalchemy.ext.asyncio import AsyncSession

    from app.db.models import (
        Account,
        ActualTransaction,
        AiConversation,
        AiMessage,
        AppUser,
        BudgetPeriod,
        Category,
        Goal,
        PdnAuditLog,
        PlanTemplateItem,
        PlannedTransaction,
        SavingsConfig,
        Subscription,
    )


    def _serialize_row(row: Any) -> dict:
        """Convert ORM row to a JSON-friendly dict.

        Datetime → ISO-8601 string. Enums → .value. Decimal → str(value).
        """
        out: dict[str, Any] = {}
        for col in row.__table__.columns:
            val = getattr(row, col.name)
            if isinstance(val, datetime):
                out[col.name] = val.isoformat()
            elif hasattr(val, "value"):  # enum
                out[col.name] = val.value
            else:
                out[col.name] = val
        return out


    async def build_export(db: AsyncSession, *, user_id: int) -> dict:
        """Return a dict with all PII for `user_id`.

        Caller MUST have called `set_tenant_scope(db, user_id)` first so RLS
        policies allow SELECT on tenant-scoped tables.
        """
        # Top-level user row (no RLS на app_user).
        user_row = await db.scalar(select(AppUser).where(AppUser.id == user_id))
        if user_row is None:
            return {}

        async def _list(model) -> list[dict]:
            result = await db.execute(select(model))
            return [_serialize_row(r) for r in result.scalars().all()]

        # Hash user_id for audit-log filter (we don't store raw user_id in audit).
        import hashlib
        user_id_hash = hashlib.sha256(str(user_id).encode("utf-8")).hexdigest()
        audit_result = await db.execute(
            select(PdnAuditLog).where(PdnAuditLog.user_id_hash == user_id_hash)
        )
        audit_rows = [_serialize_row(r) for r in audit_result.scalars().all()]

        return {
            "user": _serialize_row(user_row),
            "accounts": await _list(Account),
            "categories": await _list(Category),
            "budget_periods": await _list(BudgetPeriod),
            "plan_template_items": await _list(PlanTemplateItem),
            "planned_transactions": await _list(PlannedTransaction),
            "actual_transactions": await _list(ActualTransaction),
            "subscriptions": await _list(Subscription),
            "ai_conversations": await _list(AiConversation),
            "ai_messages": await _list(AiMessage),
            "goals": await _list(Goal),
            "savings_config": (
                _serialize_row(
                    (await db.execute(select(SavingsConfig).limit(1)))
                    .scalar_one_or_none()
                )
                if (await db.execute(select(SavingsConfig).limit(1))).scalar_one_or_none()
                else None
            ),
            "audit_log": audit_rows,
            "_meta": {
                "exported_at": datetime.utcnow().isoformat(),
                "format_version": "1.0",
            },
        }
    ```

    ```python
    """Phase 33 CMP-33-02: account deletion (soft-delete + purge).

    Soft-delete: SET `app_user.deleted_at = now()`. The user can still re-grant
    consent and resume usage within 30 days (cooling period).

    After 30 days, `purge_deleted_users_job` (worker) physically removes all
    domain rows in reverse-dependency order to avoid orphan FK violations.
    """
    from __future__ import annotations

    from datetime import datetime, timedelta, timezone

    import structlog
    from sqlalchemy import select, text
    from sqlalchemy.ext.asyncio import AsyncSession

    from app.db.models import AppUser

    logger = structlog.get_logger(__name__)

    COOLING_DAYS = 30

    # Reverse-dependency order: leaf tables first, app_user last.
    # All tenant-scoped tables list user_id FK; we delete by `user_id = $1`.
    PURGE_ORDER: list[str] = [
        "ai_message",
        "ai_conversation",
        "category_embedding",
        "actual_transaction",
        "planned_transaction",
        "subscription",
        "plan_template_item",
        "budget_period",
        "goal",
        "savings_config",
        "account",
        "category",
        # app_user last; deleted by primary key explicitly.
    ]


    async def soft_delete_account(db: AsyncSession, *, user_id: int) -> AppUser:
        """Set `deleted_at = now()` on `app_user`. Idempotent on already-deleted."""
        user = await db.scalar(select(AppUser).where(AppUser.id == user_id))
        if user is None or user.deleted_at is not None:
            return user  # may be None
        user.deleted_at = datetime.now(timezone.utc)
        await db.flush()
        return user


    async def purge_user_data(db: AsyncSession, *, user_id: int) -> dict[str, int]:
        """Cascade-delete all rows tied to `user_id`. Returns counts per table.

        Caller MUST run under `budget_admin` role (BYPASSRLS) — RLS-scoped
        SELECT/DELETE would otherwise require setting GUC per-table.
        """
        counts: dict[str, int] = {}
        for table in PURGE_ORDER:
            result = await db.execute(
                text(f"DELETE FROM {table} WHERE user_id = :uid"),
                {"uid": user_id},
            )
            counts[table] = result.rowcount or 0
        # app_user removed last.
        result = await db.execute(
            text("DELETE FROM app_user WHERE id = :uid"),
            {"uid": user_id},
        )
        counts["app_user"] = result.rowcount or 0
        return counts


    def is_due_for_purge(deleted_at: datetime, *, now: datetime | None = None) -> bool:
        if deleted_at is None:
            return False
        if now is None:
            now = datetime.now(timezone.utc)
        return deleted_at < (now - timedelta(days=COOLING_DAYS))
    ```

    ```python
    """Phase 33 CMP-33-02: purge_deleted_users_job — daily @ 02:00 MSK.

    Iterates over users with `deleted_at < now() - 30d` and performs cascade
    hard-delete via `purge_user_data`. Each user purged in an isolated
    AsyncSession transaction so a failure on one doesn't block others.

    Coordinated via pg_try_advisory_lock to prevent concurrent runs across
    worker replicas (defence-in-depth — currently single worker container).
    """
    from __future__ import annotations

    from datetime import datetime, timedelta, timezone

    import structlog
    from sqlalchemy import select, text

    from app.db.models import AppUser, PdnAuditEvent
    from app.db.session import AsyncSessionLocal
    from app.services.account_deletion import (
        COOLING_DAYS,
        is_due_for_purge,
        purge_user_data,
    )
    from app.services.pdn_audit import record_audit

    logger = structlog.get_logger(__name__)

    # Disjoint from other advisory locks (close_period=20250501, notify=20250502,
    # charge=20250503). Phase 33 = 20260101.
    ADVISORY_LOCK_KEY = 20260101


    async def purge_deleted_users_job() -> None:
        """Find users due for purge + cascade-delete + audit-event."""
        threshold = datetime.now(timezone.utc) - timedelta(days=COOLING_DAYS)
        user_ids: list[int] = []

        async with AsyncSessionLocal() as outer:
            got_lock = (
                await outer.execute(
                    text("SELECT pg_try_advisory_lock(:k)"),
                    {"k": ADVISORY_LOCK_KEY},
                )
            ).scalar_one()
            if not got_lock:
                logger.info("purge_deleted_users.skip", reason="lock_busy")
                return

            try:
                result = await outer.execute(
                    select(AppUser.id, AppUser.deleted_at).where(
                        AppUser.deleted_at.is_not(None),
                        AppUser.deleted_at < threshold,
                    )
                )
                user_ids = [row[0] for row in result.fetchall()]
                logger.info("purge_deleted_users.candidates", count=len(user_ids))
            finally:
                await outer.execute(
                    text("SELECT pg_advisory_unlock(:k)"),
                    {"k": ADVISORY_LOCK_KEY},
                )
                await outer.commit()

        for uid in user_ids:
            async with AsyncSessionLocal() as session:
                try:
                    counts = await purge_user_data(session, user_id=uid)
                    # Audit AFTER hard-delete (uses sha256 of uid).
                    await record_audit(
                        session,
                        user_id=uid,
                        event=PdnAuditEvent.deletion_completed,
                        ip=None,
                        metadata={"counts": counts},
                    )
                    await session.commit()
                    logger.info(
                        "purge_deleted_users.purged",
                        user_id=uid,
                        counts=counts,
                    )
                except Exception:
                    await session.rollback()
                    logger.exception(
                        "purge_deleted_users.failed", user_id=uid
                    )
    ```
