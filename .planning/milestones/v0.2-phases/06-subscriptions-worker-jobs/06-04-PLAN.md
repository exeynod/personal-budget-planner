    ```python
    import logging
    from datetime import date, timedelta
    from aiogram import Bot
    from sqlalchemy import select, text
    from app.db.session import AsyncSessionLocal
    from app.db.models import Subscription, AppUser
    from app.core.settings import settings
    from app.services.periods import _today_in_app_tz

    logger = logging.getLogger(__name__)
    ADVISORY_LOCK_KEY = 20250502  # per close_period.py:42 comment

    def _format_amount_rub(cents: int) -> str:
        rub = cents / 100
        return f"{rub:,.2f}".replace(",", " ").replace(".", ",")

    async def notify_subscriptions_job():
        async with AsyncSessionLocal() as db:
            got = await db.scalar(text("SELECT pg_try_advisory_lock(:k)"), {"k": ADVISORY_LOCK_KEY})
            if not got:
                logger.info("notify_subscriptions: lock busy, skip")
                return
            bot = None
            try:
                user = await db.scalar(select(AppUser).limit(1))
                if user is None or user.tg_chat_id is None:
                    logger.info("notify_subscriptions: no tg_chat_id, skip")
                    return
                today = _today_in_app_tz()
                # Все активные подписки, у которых today + notify_days_before == next_charge_date
                # Эквивалент: next_charge_date - today = notify_days_before
                rows = (await db.execute(
                    select(Subscription).where(Subscription.is_active == True)
                )).scalars().all()
                due = [s for s in rows if (s.next_charge_date - today).days == s.notify_days_before]
                if not due:
                    logger.info("notify_subscriptions: nothing to notify")
                    return
                bot = Bot(token=settings.BOT_TOKEN)
                for sub in due:
                    text_msg = (
                        f"🔔 Подписка «{sub.name}»\n"
                        f"   Спишется {_format_amount_rub(sub.amount_cents)} ₽ "
                        f"через {sub.notify_days_before} дн. "
                        f"({sub.next_charge_date.strftime('%d.%m')})"
                    )
                    try:
                        await bot.send_message(chat_id=user.tg_chat_id, text=text_msg)
                        logger.info("notify_subscriptions: sent for sub_id=%s", sub.id)
                    except Exception as e:
                        logger.exception("notify_subscriptions: send failed sub_id=%s: %s", sub.id, e)
            except Exception as e:
                logger.exception("notify_subscriptions: failed: %s", e)
            finally:
                if bot is not None:
                    await bot.session.close()
                await db.execute(text("SELECT pg_advisory_unlock(:k)"), {"k": ADVISORY_LOCK_KEY})
    ```

    ```python
    import logging
    from sqlalchemy import select, text
    from app.db.session import AsyncSessionLocal
    from app.db.models import Subscription, AppUser
    from app.services.periods import _today_in_app_tz
    from app.services.settings import get_cycle_start_day
    from app.services.subscriptions import charge_subscription, AlreadyChargedError

    logger = logging.getLogger(__name__)
    ADVISORY_LOCK_KEY = 20250503

    async def charge_subscriptions_job():
        async with AsyncSessionLocal() as db_outer:
            got = await db_outer.scalar(text("SELECT pg_try_advisory_lock(:k)"), {"k": ADVISORY_LOCK_KEY})
            if not got:
                logger.info("charge_subscriptions: lock busy")
                return
            try:
                user = await db_outer.scalar(select(AppUser).limit(1))
                if user is None:
                    logger.info("charge_subscriptions: no user, skip")
                    return
                today = _today_in_app_tz()
                cycle_start = await get_cycle_start_day(db_outer, user.tg_user_id)
                ids = (await db_outer.execute(
                    select(Subscription.id).where(
                        Subscription.is_active == True,
                        Subscription.next_charge_date == today,
                    )
                )).scalars().all()
                logger.info("charge_subscriptions: %d subs due today", len(ids))
                for sub_id in ids:
                    # отдельная сессия на каждую подписку — изоляция rollback
                    async with AsyncSessionLocal() as db:
                        try:
                            planned, new_date = await charge_subscription(db, sub_id, cycle_start_day=cycle_start)
                            await db.commit()
                            logger.info("charge_subscriptions: sub_id=%s charged, next=%s", sub_id, new_date)
                        except AlreadyChargedError:
                            logger.warning("charge_subscriptions: sub_id=%s already charged today, skip", sub_id)
                        except Exception as e:
                            logger.exception("charge_subscriptions: sub_id=%s failed: %s", sub_id, e)
            finally:
                await db_outer.execute(text("SELECT pg_advisory_unlock(:k)"), {"k": ADVISORY_LOCK_KEY})
    ```
