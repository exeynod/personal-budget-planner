**2. Создать `app/bot/api_client.py`:**

```python
"""Bot → API internal client.

Used by /start handler (and Phase 4+ command handlers) to call
/api/v1/internal/* endpoints with X-Internal-Token. The bot is the only
caller that has the INTERNAL_TOKEN secret.
"""
import httpx
import structlog

from app.core.settings import settings


logger = structlog.get_logger(__name__)


class InternalApiError(Exception):
    """Raised when the internal API call fails (network or non-2xx response)."""


async def bind_chat_id(*, tg_user_id: int, tg_chat_id: int) -> None:
    """POST /api/v1/internal/telegram/chat-bind.

    Args:
        tg_user_id: Telegram user id (must equal OWNER_TG_ID — checked by caller).
        tg_chat_id: Telegram chat id to persist for push notifications.

    Raises:
        InternalApiError: on connection failure or non-2xx response.
    """
    url_path = "/api/v1/internal/telegram/chat-bind"
    payload = {"tg_user_id": tg_user_id, "tg_chat_id": tg_chat_id}
    headers = {"X-Internal-Token": settings.INTERNAL_TOKEN}

    try:
        async with httpx.AsyncClient(
            base_url=settings.API_BASE_URL,
            timeout=5.0,
        ) as client:
            response = await client.post(url_path, json=payload, headers=headers)
            response.raise_for_status()
    except httpx.HTTPError as exc:
        logger.warning(
            "bot.chat_bind.failed",
            tg_user_id=tg_user_id,
            error=str(exc),
        )
        raise InternalApiError(f"chat-bind failed: {exc}") from exc

    logger.info("bot.chat_bind.ok", tg_user_id=tg_user_id, tg_chat_id=tg_chat_id)
```

**3. Создать `app/bot/handlers.py`:**

```python
"""aiogram /start handler with chat-bind + WebApp button (ONB-03).

This module replaces the Phase 1 stub in main_bot.py.

Behavior:
- /start from non-OWNER → "Бот приватный" (defence-in-depth; Telegram-level
  ACL via private bot already filters most cases, but we re-check here per
  HLD §5).
- /start from OWNER:
  1. Calls internal /telegram/chat-bind to persist tg_chat_id.
  2. Parses optional payload (`/start onboard` → friendlier copy).
  3. Replies with greeting + InlineKeyboardButton(web_app=WebAppInfo(url=MINI_APP_URL)).
- If chat-bind fails (network), logs a warning and still replies with the
  WebApp button — user can re-issue /start to retry binding.
"""
import structlog
from aiogram import Router
from aiogram.filters import CommandObject, CommandStart
from aiogram.types import (
    InlineKeyboardButton,
    InlineKeyboardMarkup,
    Message,
    WebAppInfo,
)

from app.bot.api_client import InternalApiError, bind_chat_id
from app.core.settings import settings


logger = structlog.get_logger(__name__)
router = Router()


def _open_app_keyboard() -> InlineKeyboardMarkup:
    btn = InlineKeyboardButton(
        text="Открыть бюджет",
        web_app=WebAppInfo(url=settings.MINI_APP_URL),
    )
    return InlineKeyboardMarkup(inline_keyboard=[[btn]])


@router.message(CommandStart())
async def cmd_start(message: Message, command: CommandObject) -> None:
    """/start — chat-bind for OWNER, otherwise reject."""
    if not message.from_user:
        return  # safety: ignore service messages

    user_id = message.from_user.id

    # OWNER_TG_ID whitelist (HLD §5, AUTH-02 carry-over)
    if user_id != settings.OWNER_TG_ID:
        await message.answer("Бот приватный.")
        logger.info("bot.start.rejected_non_owner", tg_user_id=user_id)
        return

    chat_id = message.chat.id

    # Step 1: bind tg_chat_id (best-effort; user can re-issue /start to retry)
    try:
        await bind_chat_id(tg_user_id=user_id, tg_chat_id=chat_id)
        chat_bound = True
    except InternalApiError:
        # Already logged inside api_client. Continue and still send the button.
        chat_bound = False

    # Step 2: parse deep-link payload (Pattern 1)
    payload = command.args  # str or None ("onboard" if from Mini App link)

    if payload == "onboard":
        greeting = (
            "Готово, push-уведомления включены.\n"
            "Откройте Mini App для настройки бюджета."
        )
    elif chat_bound:
        greeting = (
            "Бот запущен и готов к работе.\n"
            "Откройте Mini App, чтобы управлять бюджетом."
        )
    else:
        greeting = (
            "Бот запущен, но не удалось привязать чат для уведомлений.\n"
            "Попробуйте /start ещё раз через минуту.\n"
            "Можно открыть Mini App — push-уведомления подключатся позже."
        )

    await message.answer(greeting, reply_markup=_open_app_keyboard())
    logger.info(
        "bot.start.replied",
        tg_user_id=user_id,
        tg_chat_id=chat_id,
        chat_bound=chat_bound,
        payload=payload,
    )
```

```python
"""aiogram bot entry point for the bot container.

Phase 2 scope (ONB-03):
- ``/start`` handler: OWNER-only, calls /api/v1/internal/telegram/chat-bind to
  persist tg_chat_id, replies with InlineKeyboardButton(web_app=WebAppInfo).
  Implementation lives in ``app.bot.handlers`` (factored out for testability
  and to keep main_bot.py minimal).
- ``GET /healthz`` on port 8001 (INF-05, D-12) via aiohttp running concurrently
  with ``dp.start_polling`` in the same event loop.

Real command handlers (/add, /income, /balance, /today, /app) land in Phase 4.
Mode: long-poll (D-04).
"""
import asyncio

import structlog
from aiogram import Bot, Dispatcher
from aiogram.client.default import DefaultBotProperties
from aiogram.enums import ParseMode
from aiohttp import web

from app.bot.handlers import router  # Phase 2 handler — replaces stub
from app.core.logging import configure_logging
from app.core.settings import settings


configure_logging(settings.LOG_LEVEL, settings.LOG_FORMAT)
logger = structlog.get_logger(__name__)


async def health_handler(request: web.Request) -> web.Response:
    """``GET /healthz`` for the bot container (INF-05, port 8001)."""
    return web.Response(
        text='{"status":"ok","service":"bot"}',
        content_type="application/json",
    )


async def main() -> None:
    bot = Bot(
        token=settings.BOT_TOKEN,
        default=DefaultBotProperties(parse_mode=ParseMode.HTML),
    )
    dp = Dispatcher()
    dp.include_router(router)

    # Start aiohttp healthz server on port 8001 (D-12, INF-05).
    health_app = web.Application()
    health_app.router.add_get("/healthz", health_handler)
    runner = web.AppRunner(health_app)
    await runner.setup()
    site = web.TCPSite(runner, "0.0.0.0", 8001)
    await site.start()
    logger.info("bot.healthz.started", port=8001)

    # Long-polling (D-04). Blocks until shutdown.
    logger.info("bot.polling.started")
    try:
        await dp.start_polling(bot)
    finally:
        await runner.cleanup()
        await bot.session.close()
        logger.info("bot.shutdown")


if __name__ == "__main__":
    asyncio.run(main())
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Telegram → bot | Telegram гарантирует подлинность message.from_user.id |
| bot → API internal | bot имеет INTERNAL_TOKEN secret; X-Internal-Token проверяется FastAPI |
| bot → Telegram WebApp | WebAppInfo URL должен быть HTTPS в prod (Telegram requirement) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-bot-non-owner | Spoofing | cmd_start | mitigate | OWNER_TG_ID check возвращает "Бот приватный" не выполняя chat-bind; defence-in-depth поверх Telegram-уровневого ACL (приватный бот) |
| T-bot-token-leak | Info Disclosure | api_client | mitigate | INTERNAL_TOKEN не логируется (structlog поля не включают token) |
| T-bot-conn-error | Denial | bind_chat_id | mitigate | InternalApiError caught в handler; user всё равно получает WebApp button + сообщение «попробуйте снова через минуту» |
| T-webapp-url-injection | Tampering | WebAppInfo | accept | URL берётся из settings.MINI_APP_URL (config), не из user input — нет вектора инъекции |
| T-payload-injection | Tampering | command.args | mitigate | Парсится только как '== "onboard"' equality check; никаких eval/exec; даже если payload содержит SQL/HTML — он не достигает БД и не рендерится в HTML (отправляется только статичная текстовая копия) |
| T-bot-replay | Replay | cmd_start | accept | Telegram гарантирует unique message.message_id; повторный /start корректен (chat-bind idempotent через UPSERT) |
</threat_model>
