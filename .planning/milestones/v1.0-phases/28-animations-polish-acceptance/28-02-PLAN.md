```swift
//  PosterAnimationsAuditTests.swift — Phase 28 POL-01/POL-02/POL-03 (iOS)
//  Verifies all 11 PosterAnimations curves are instantiable and that
//  the posterAnimation modifier honors accessibilityReduceMotion.

import XCTest
import SwiftUI
@testable import BudgetPlanner

final class PosterAnimationsAuditTests: XCTestCase {

    func test_all_11_animations_instantiable() throws {
        // Sanity: каждая из 11 declared anims существует и не nil.
        _ = PosterAnimations.posterRowIn(delay: 0.1)
        _ = PosterAnimations.posterRiseIn(delay: 0)
        _ = PosterAnimations.posterBarFill(delay: 0, duration: 0.7)
        _ = PosterAnimations.posterTabPop
        _ = PosterAnimations.posterPopIn(delay: 0)
        _ = PosterAnimations.posterCheck
        _ = PosterAnimations.posterDot
        _ = PosterAnimations.slideInFwdTransition()
        _ = PosterAnimations.slideInBackTransition()
        _ = PosterAnimations.posterSlide
        _ = PosterAnimations.posterTabSwap
        _ = PosterAnimations.posterToastIn

        XCTAssertEqual(PosterAnimations.toastLifeMs, 1700, "Toast lifetime must equal web --poster-toast-life")
    }

    func test_stagger_formulas_per_design_system_7_4() throws {
        // §7.4: rows = 0.08 + i*0.045
        XCTAssertEqual(PosterAnimations.rowStagger(i: 0), 0.08, accuracy: 1e-6)
        XCTAssertEqual(PosterAnimations.rowStagger(i: 1), 0.125, accuracy: 1e-6)
        // day groups = 0.05 + i*0.07
        XCTAssertEqual(PosterAnimations.dayGroupStagger(i: 0), 0.05, accuracy: 1e-6)
        XCTAssertEqual(PosterAnimations.dayGroupStagger(i: 2), 0.19, accuracy: 1e-6)
        // hints = 0.18 + i*0.08
        XCTAssertEqual(PosterAnimations.hintStagger(i: 3), 0.42, accuracy: 1e-6)
        // regulars = 0.32 + i*0.09
        XCTAssertEqual(PosterAnimations.regularStagger(i: 1), 0.41, accuracy: 1e-6)
    }

    func test_dot_phase_offset_per_dot() throws {
        XCTAssertEqual(PosterAnimations.dotPhase(i: 0), 0.0, accuracy: 1e-6)
        XCTAssertEqual(PosterAnimations.dotPhase(i: 1), 0.18, accuracy: 1e-6)
        XCTAssertEqual(PosterAnimations.dotPhase(i: 2), 0.36, accuracy: 1e-6)
    }

    /// Confirms posterAnimation modifier exists and accepts the canonical signature.
    /// (Behavior under reduce-motion is exercised in SwiftUI Preview / Simulator manual smoke;
    /// the env override cannot be programmatically toggled in plain XCTest without a host app.)
    func test_posterAnimation_modifier_compiles_with_canonical_signature() throws {
        let probe = Color.red.posterAnimation(PosterAnimations.posterRiseIn(), value: 0)
        XCTAssertNotNil(probe)
    }

    func test_posterTransition_modifier_compiles() throws {
        let probe = Color.red.posterTransition(PosterAnimations.slideInFwdTransition())
        XCTAssertNotNil(probe)
    }
}
```
