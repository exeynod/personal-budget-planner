       ```swift
       enum AnalyticsData {
         static let MONTHS_RU_SHORT = ["ЯНВ","ФЕВ","МАР","АПР","МАЙ","ИЮН",
           "ИЮЛ","АВГ","СЕН","ОКТ","НОЯ","ДЕК"]

         enum GroupMode: Equatable { case day, week, cat }

         struct MonthOption: Equatable, Identifiable {
           let label: String
           let year: Int
           let month: Int
           let periodStart: String  // YYYY-MM-01
           let periodEnd: String    // YYYY-MM-DD last day
           var id: String { periodStart }
         }

         static func lastNMonths(_ now: Date, _ n: Int, calendar: Calendar = .current) -> [MonthOption] {
           var out: [MonthOption] = []
           for i in stride(from: n - 1, through: 0, by: -1) {
             let comps = calendar.dateComponents([.year, .month], from: now)
             let year = comps.year ?? 2026
             let month = comps.month ?? 1
             let total = (year * 12 + (month - 1)) - i
             let y = total / 12
             let m = (total % 12) + 1
             let yy = String(y % 100)
             let lastDay = calendar.range(of: .day, in: .month,
               for: calendar.date(from: DateComponents(year: y, month: m, day: 1))!)!.upperBound - 1
             out.append(MonthOption(
               label: "\(MONTHS_RU_SHORT[m-1]) \(yy)",
               year: y, month: m,
               periodStart: String(format: "%04d-%02d-01", y, m),
               periodEnd: String(format: "%04d-%02d-%02d", y, m, lastDay)
             ))
           }
           return out
         }

         static func groupByDay(_ actuals: [ActualV10DTO], periodStart: Date, periodEnd: Date)
           -> [(key: Date, sumCents: Int)]
         {
           let filtered = actuals.filter { $0.kind == .expense && $0.txDate >= periodStart && $0.txDate <= periodEnd }
           let grouped = Dictionary(grouping: filtered, by: { $0.txDate })
           return grouped.map { (key: $0.key, sumCents: $0.value.reduce(0) { $0 + abs($1.amountCents) }) }
             .sorted { $0.key < $1.key }
         }

         static func groupByWeek(_ actuals: [ActualV10DTO], calendar: Calendar = .current)
           -> [(weekIdx: Int, sumCents: Int)]
         {
           let filtered = actuals.filter { $0.kind == .expense }
           var buckets: [Int: Int] = [:]
           for t in filtered {
             let day = calendar.component(.day, from: t.txDate)
             let weekIdx = (day + 6) / 7  // 1..5
             buckets[weekIdx, default: 0] += abs(t.amountCents)
           }
           return buckets.map { (weekIdx: $0.key, sumCents: $0.value) }.sorted { $0.weekIdx < $1.weekIdx }
         }

         static func groupByCategory(_ actuals: [ActualV10DTO], categories: [CategoryV10DTO])
           -> [(categoryId: Int, name: String, planCents: Int, sumCents: Int)]
         {
           let filtered = actuals.filter { $0.kind == .expense }
           let catMap = Dictionary(uniqueKeysWithValues: categories.map { ($0.id, $0) })
           var sums: [Int: Int] = [:]
           for t in filtered {
             guard let cid = t.categoryId else { continue }
             sums[cid, default: 0] += abs(t.amountCents)
           }
           return sums.map { id, sum in
             let cat = catMap[id]
             return (categoryId: id, name: cat?.name ?? "?", planCents: cat?.planCents ?? 0, sumCents: sum)
           }.sorted { $0.sumCents > $1.sumCents }
         }

         static func computeKPISpent(curr: [ActualV10DTO], prev: [ActualV10DTO])
           -> (sumCents: Int, deltaCents: Int, deltaPct: Int)
         {
           let sum: ([ActualV10DTO]) -> Int = { xs in
             xs.filter { $0.kind == .expense }.reduce(0) { $0 + abs($1.amountCents) }
           }
           let curSum = sum(curr); let prevSum = sum(prev)
           let delta = curSum - prevSum
           let pct = prevSum == 0 ? 0 : Int((Double(delta) / Double(prevSum) * 100.0).rounded())
           return (curSum, delta, pct)
         }

         static func computeKPISaved(actuals: [ActualV10DTO], categories: [CategoryV10DTO]) -> Int {
           let facts = groupByCategory(actuals, categories: categories)
           let factById = Dictionary(uniqueKeysWithValues: facts.map { ($0.categoryId, $0.sumCents) })
           var saved = 0
           for c in categories {
             if c.code == "savings" || c.paused { continue }
             let fact = factById[c.id] ?? 0
             saved += max(0, (c.planCents ?? 0) - fact)
           }
           return saved
         }

         static func shouldHighlightRed(barSum: Int, barPlan: Int, threshold: Double = 0.75) -> Bool {
           guard barPlan > 0 else { return false }
           return Double(barSum) / Double(barPlan) >= threshold
         }

         static func computePct(sum: Int, plan: Int) -> Int {
           guard plan > 0 else { return 0 }
           return max(0, min(100, Int((Double(sum) / Double(plan) * 100).rounded())))
         }
       }
       ```

       ```swift
       @MainActor @Observable final class AnalyticsV10ViewModel {
         enum Status: Equatable { case idle, loading, ready, error(String) }
         var status: Status = .idle
         var monthOptions: [AnalyticsData.MonthOption] = []
         var selectedMonth: AnalyticsData.MonthOption? = nil
         var groupMode: AnalyticsData.GroupMode = .day
         var actuals: [ActualV10DTO] = []
         var prevActuals: [ActualV10DTO] = []
         var categories: [CategoryV10DTO] = []
         var topCats: [TopCategoryItemDTO] = []
         private var inFlight: Bool = false

         init() {
           monthOptions = AnalyticsData.lastNMonths(Date(), 3)
           selectedMonth = monthOptions.last
         }

         func selectMonth(_ m: AnalyticsData.MonthOption) async {
           selectedMonth = m
           await load()
         }

         func selectGroup(_ m: AnalyticsData.GroupMode) {
           groupMode = m
         }

         func load() async {
           guard let m = selectedMonth else { return }
           if inFlight { return }
           inFlight = true; defer { inFlight = false }
           status = .loading
           do {
             async let catsTask = CategoriesV10API.list()
             async let topsTask = AnalyticsAPI.topCategories(periodStart: m.periodStart, periodEnd: m.periodEnd, limit: 5)
             async let actsTask = ActualV10API.list(periodStart: m.periodStart, periodEnd: m.periodEnd)
             // Prev month for delta:
             let prevIdx = (monthOptions.firstIndex(of: m) ?? 0) - 1
             let prevTask: Task<[ActualV10DTO], Error>?
             if prevIdx >= 0 {
               let pm = monthOptions[prevIdx]
               prevTask = Task { try await ActualV10API.list(periodStart: pm.periodStart, periodEnd: pm.periodEnd) }
             } else {
               prevTask = nil
             }
             let (cats, tops, acts) = try await (catsTask, topsTask, actsTask)
             categories = cats; topCats = tops; actuals = acts
             prevActuals = (try? await prevTask?.value) ?? []
             status = .ready
           } catch {
             status = .error("Не удалось загрузить аналитику")
           }
         }

         var kpiSpent: (sumCents: Int, deltaCents: Int, deltaPct: Int) {
           AnalyticsData.computeKPISpent(curr: actuals, prev: prevActuals)
         }
         var kpiSaved: Int {
           AnalyticsData.computeKPISaved(actuals: actuals, categories: categories)
         }
       }
       ```
