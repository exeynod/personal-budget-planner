       ```swift
       struct AccountCreateRequest: Encodable {
         let bank: String
         let kind: AccountKind
         let mask: String?
         let balanceCents: Int
         let primary: Bool?
         func encode(to encoder: Encoder) throws {
           var c = encoder.container(keyedBy: CodingKeys.self)
           try c.encode(bank, forKey: .bank)
           try c.encode(kind, forKey: .kind)
           try c.encodeIfPresent(mask, forKey: .mask)
           try c.encode(balanceCents, forKey: .balanceCents)
           try c.encodeIfPresent(primary, forKey: .primary)
         }
         enum CodingKeys: String, CodingKey { case bank, kind, mask, balanceCents, primary }
       }

       extension AccountsAPI {
         static func create(_ req: AccountCreateRequest) async throws -> AccountDTO {
           try await APIClient.shared.request("POST", "/api/v1/accounts", body: req)
         }
       }
       ```

    2. **`AccountsData.swift`** (NEW):
       ```swift
       import Foundation

       enum AccountsData {
         static func sumBalances(_ list: [AccountDTO]) -> Int {
           list.reduce(0) { $0 + $1.balanceCents }
         }
         static func count(_ list: [AccountDTO]) -> Int { list.count }

         static func formatBankSubtitle(_ a: AccountDTO) -> String {
           switch a.kind {
           case .cash: return "наличные"
           case .savings: return "накопит. счёт"
           case .card: return a.mask.map { "карта ·· \($0)" } ?? "карта"
           }
         }

         static func filterByAccount(_ actuals: [ActualV10DTO], accountId: Int) -> [ActualV10DTO] {
           actuals.filter { $0.accountId == accountId }
         }

         static func sumPeriodOps(_ actuals: [ActualV10DTO], periodStart: Date, periodEnd: Date)
           -> (count: Int, sumCents: Int)
         {
           let inRange = actuals.filter { $0.txDate >= periodStart && $0.txDate <= periodEnd }
           return (inRange.count, inRange.reduce(0) { $0 + abs($1.amountCents) })
         }

         static func isValidNewAccountDraft(bank: String, balanceCents: Int) -> Bool {
           !bank.trimmingCharacters(in: .whitespaces).isEmpty && balanceCents >= 0
         }
       }
       ```

       ```swift
         var sheet: SheetMode = .none

         func load() async {
           if inFlight { return }
       ```
