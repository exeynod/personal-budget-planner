       """Rule-engine для GET /ai/observation (AI-V10-03).

       Pure Python — никаких LLM-вызовов. Кэш per-user 1 час in-memory.
       """
       from dataclasses import dataclass
       from datetime import datetime, timedelta
       from typing import Optional
       from zoneinfo import ZoneInfo  # stdlib 3.9+
       from sqlalchemy.ext.asyncio import AsyncSession

       MSK = ZoneInfo("Europe/Moscow")
       CACHE_TTL = timedelta(hours=1)

       @dataclass
       class ObservationResult:
           text: str
           generated_at: datetime

       # In-memory per-user cache. Key = user_id, value = ObservationResult.
       # Cleared on process restart — acceptable since this is single-tenant pet.
       OBSERVATION_CACHE: dict[int, ObservationResult] = {}

       MONTHS_RU_NOM = ["январь","февраль","март","апрель","май","июнь",
                         "июль","август","сентябрь","октябрь","ноябрь","декабрь"]

       async def build_observation(
           db: AsyncSession, *, user_id: int, now: Optional[datetime] = None,
       ) -> ObservationResult:
           """Returns cached observation or computes a new one applying rule priority."""
           now = now or datetime.now(MSK)
           cached = OBSERVATION_CACHE.get(user_id)
           if cached and (now - cached.generated_at) < CACHE_TTL:
               return cached
           text = await _compute_text(db, user_id=user_id, now=now)
           result = ObservationResult(text=text, generated_at=now)
           OBSERVATION_CACHE[user_id] = result
           return result

       async def _compute_text(db: AsyncSession, *, user_id: int, now: datetime) -> str:
           # Priority 1: over-limit category (fact_cents > plan_cents).
           # Query: SELECT cat.name, cat.plan_cents, SUM(|amt|) AS fact
           #   FROM category cat JOIN actual_transaction tx ON tx.category_id = cat.id
           #   WHERE cat.user_id = :uid AND cat.code != 'savings' AND NOT cat.paused
           #     AND tx.kind = 'expense'
           #     AND tx.tx_date >= date_trunc('month', :now)
           #   GROUP BY cat.id, cat.name, cat.plan_cents
           #   HAVING SUM(|amt|) > cat.plan_cents
           #   ORDER BY (SUM(|amt|)::float / NULLIF(cat.plan_cents,0)) DESC
           #   LIMIT 1
           # Если есть — return f"{name} уже +{pct}% к лимиту".

           # Priority 2: subs charge tomorrow (cycle='monthly', day_of_month == (now+1d).day).
           # Query: SELECT SUM(amount_cents) FROM subscription
           #   WHERE user_id = :uid AND is_active AND cycle='monthly' AND day_of_month = :tomorrow_day
           # Если sum > 0: return f"Завтра списание подписок на {sum/100} ₽".

           # Priority 3: week savings (kind in ('roundup','deposit') за последние 7 дней).
           # Query: SELECT SUM(|amt|) FROM actual_transaction
           #   WHERE user_id = :uid AND kind IN ('roundup','deposit')
           #     AND tx_date >= :now - INTERVAL '7 days'
           # Если sum > 0: return f"За неделю экономия {sum/100} ₽".

           # Priority 4: month surplus (User.income_cents - SUM expense fact > 0).
           # Query: SELECT u.income_cents - COALESCE(SUM(|amt|),0) FROM app_user u
           #   LEFT JOIN actual_transaction tx ON tx.user_id = u.id
           #     AND tx.kind = 'expense' AND tx.tx_date >= date_trunc('month', :now)
           #   WHERE u.id = :uid GROUP BY u.income_cents
           # Если result > 0: return f"{MONTHS_RU_NOM[month-1].capitalize()} в плюсе на {result/100} ₽".

           # Fallback: "Веди учёт регулярно — {today}", today = "9 мая" (день + month_genitive_ru).
           ...
