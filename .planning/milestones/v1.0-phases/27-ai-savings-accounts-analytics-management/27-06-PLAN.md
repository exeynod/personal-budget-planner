    3. **`MgmtHubMount.tsx`** (NEW, ~100 LOC):
       ```tsx
       import { useEffect, useState } from 'react';
       import { fetchMeV10 } from '../../api/me';  // verify exact name; should exist from Phase 24
       import { usePosterRouter } from '../common';
       import { MgmtHubView } from './MgmtHubView';
       import { PlanMount } from '../Plan';
       import { AccountsListMount } from '../Accounts';
       import { AnalyticsMount } from '../Analytics';
       import { SettingsMount } from './SettingsMount';
       import { AccessMount } from './AccessMount';

       export function MgmtHubMount() {
         const router = usePosterRouter();
         const [isOwner, setIsOwner] = useState(false);

         useEffect(() => {
           let cancelled = false;
           fetchMeV10()
             .then((me) => {
               if (cancelled) return;
               setIsOwner(me.role === 'owner');
             })
             .catch(() => { /* default false */ });
           return () => { cancelled = true; };
         }, []);

         function handleRowTap(id: MgmtRowId) {
           if (id === 'plan') router.push(<PlanMount />);
           else if (id === 'accounts') router.push(<AccountsListMount />);
           else if (id === 'analytics') router.push(<AnalyticsMount />);
           else if (id === 'settings') router.push(<SettingsMount />);
           else if (id === 'access') router.push(<AccessMount />);
         }

         return (
           <MgmtHubView
             isOwner={isOwner}
             onRowTap={handleRowTap}
             canPop={router.canPop}
             onBack={() => router.pop()}
           />
         );
       }
       ```

       **NB**: Verify `fetchMeV10` import path/name; Phase 24 plan 24-10 should have created it. If not, fall back to v0.6 `getMe()` from api/me.ts (cast role).
