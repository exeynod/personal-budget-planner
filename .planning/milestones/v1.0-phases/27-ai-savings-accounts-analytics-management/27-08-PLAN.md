       enum SavingsAPI {
         static func summary() async throws -> SavingsSummaryDTO {
           try await APIClient.shared.request("GET", "/api/v1/savings")
         }
         static func patchConfig(roundupEnabled: Bool? = nil, roundupBase: Int? = nil) async throws -> SavingsConfigDTO {
           struct Body: Encodable {
             let roundupEnabled: Bool?
             let roundupBase: Int?
             func encode(to encoder: Encoder) throws {
               var c = encoder.container(keyedBy: CodingKeys.self)
               try c.encodeIfPresent(roundupEnabled, forKey: .roundupEnabled)
               try c.encodeIfPresent(roundupBase, forKey: .roundupBase)
             }
             enum CodingKeys: String, CodingKey { case roundupEnabled, roundupBase }
           }
           return try await APIClient.shared.request(
             "PATCH", "/api/v1/savings/config",
             body: Body(roundupEnabled: roundupEnabled, roundupBase: roundupBase)
           )
         }
         static func postDeposit(amountCents: Int, accountId: Int?, goalId: Int?) async throws -> DepositResponseDTO {
           struct Body: Encodable {
             let amountCents: Int
             let accountId: Int?
             let goalId: Int?
             func encode(to encoder: Encoder) throws {
               var c = encoder.container(keyedBy: CodingKeys.self)
               try c.encode(amountCents, forKey: .amountCents)
               try c.encodeIfPresent(accountId, forKey: .accountId)
               try c.encodeIfPresent(goalId, forKey: .goalId)
             }
             enum CodingKeys: String, CodingKey { case amountCents, accountId, goalId }
           }
           return try await APIClient.shared.request(
             "POST", "/api/v1/savings/deposit",
             body: Body(amountCents: amountCents, accountId: accountId, goalId: goalId)
           )
         }
       }
       ```

       import Foundation

       enum SavingsData {
         static let MONTHS_RU_GEN = ["января","февраля","марта","апреля","мая","июня",
           "июля","августа","сентября","октября","ноября","декабря"]

         static func computeProgressPct(currentCents: Int, targetCents: Int) -> Int {
           guard targetCents > 0 else { return 0 }
           let pct = Int((Double(currentCents) / Double(targetCents) * 100.0).rounded())
           return max(0, min(100, pct))
         }

         static func formatDueRu(_ d: Date?, calendar: Calendar = .current) -> String? {
           guard let d = d else { return nil }
           let comps = calendar.dateComponents([.day, .month, .year], from: d)
           guard let day = comps.day, let mon = comps.month, let year = comps.year else { return nil }
           guard mon >= 1, mon <= 12 else { return nil }
           return "до \(day) \(MONTHS_RU_GEN[mon - 1]) \(year)"
         }

         static func isValidGoalDraft(name: String, targetCents: Int) -> Bool {
           return !name.trimmingCharacters(in: .whitespaces).isEmpty && targetCents > 0
         }

         static func isValidDepositDraft(amountCents: Int, accountId: Int?) -> Bool {
           return amountCents > 0 && accountId != nil
         }
       }
       ```

       ```swift
       import Foundation
       import Observation

       @MainActor
       @Observable
       final class SavingsV10ViewModel {
         enum Status: Equatable { case idle, loading, ready, error(String) }
         enum SheetMode: Equatable { case none, newGoal, deposit }

         var status: Status = .idle
         var snapshot: SavingsSummaryDTO? = nil
         var accounts: [AccountDTO] = []
         var sheet: SheetMode = .none
         var submitting: Bool = false
         private var inFlight: Bool = false

         func load() async {
           if inFlight { return }
           inFlight = true
           defer { inFlight = false }
           status = .loading
           do {
             async let snapTask = SavingsAPI.summary()
             async let accsTask = AccountsAPI.list()
             let (snap, accs) = try await (snapTask, accsTask)
             snapshot = snap
             accounts = accs
             status = .ready
           } catch {
             status = .error("Не удалось загрузить копилку")
           }
         }

         func toggleRoundup(_ enabled: Bool) async {
           guard var snap = snapshot else { return }
           // Optimistic.
           snap = SavingsSummaryDTO(
             totalCents: snap.totalCents, monthInCents: snap.monthInCents,
             config: SavingsConfigDTO(roundupEnabled: enabled, roundupBase: snap.config.roundupBase),
             goals: snap.goals
           )
           snapshot = snap
           do {
             let cfg = try await SavingsAPI.patchConfig(roundupEnabled: enabled)
             // Reflect server response.
             if let s = snapshot {
               snapshot = SavingsSummaryDTO(
                 totalCents: s.totalCents, monthInCents: s.monthInCents, config: cfg, goals: s.goals
               )
             }
           } catch {
             // Silent — Phase 28 polish wires toast. Reload to recover.
             await load()
           }
         }

         func selectBase(_ base: Int) async {
           guard let snap = snapshot else { return }
           snapshot = SavingsSummaryDTO(
             totalCents: snap.totalCents, monthInCents: snap.monthInCents,
             config: SavingsConfigDTO(roundupEnabled: snap.config.roundupEnabled, roundupBase: base),
             goals: snap.goals
           )
           do {
             let cfg = try await SavingsAPI.patchConfig(roundupBase: base)
             if let s = snapshot {
               snapshot = SavingsSummaryDTO(
                 totalCents: s.totalCents, monthInCents: s.monthInCents, config: cfg, goals: s.goals
               )
             }
           } catch { await load() }
         }

         func createGoal(name: String, targetCents: Int, due: Date?) async {
           submitting = true
           defer { submitting = false }
           do {
             _ = try await GoalsAPI.create(GoalCreateRequest(name: name, targetCents: targetCents, due: due))
             sheet = .none
             await load()
           } catch { /* silent */ }
         }

         func deposit(amountCents: Int, accountId: Int, goalId: Int?) async {
           submitting = true
           defer { submitting = false }
           do {
             _ = try await SavingsAPI.postDeposit(amountCents: amountCents, accountId: accountId, goalId: goalId)
             sheet = .none
             await load()
           } catch { /* silent */ }
         }
       }
       ```
