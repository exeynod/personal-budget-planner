       """Pydantic v2 schemas for /api/v1/plan-month (Phase 26, PLAN-V10-06)."""
       from pydantic import BaseModel, ConfigDict, Field, model_validator
       from app.api.schemas.categories import CategoryRead

       class PlanMonthItem(BaseModel):
           model_config = ConfigDict(extra="forbid")
           category_id: int = Field(gt=0)
           plan_cents: int = Field(ge=0)

       class PlanMonthPatch(BaseModel):
           model_config = ConfigDict(extra="forbid")
           plans: list[PlanMonthItem] = Field(min_length=1)

           @model_validator(mode="after")
           def _no_duplicate_ids(self) -> "PlanMonthPatch":
               ids = [p.category_id for p in self.plans]
               if len(ids) != len(set(ids)):
                   raise ValueError("duplicate category_id in plans list")
               return self

       class PlanMonthResponse(BaseModel):
           categories: list[CategoryRead]

    2. **Создать `app/services/plan_month.py`** (~80 LOC):
       ```python
       """Atomic plan-month update service (Phase 26, PLAN-V10-06)."""
       from sqlalchemy import select
       from sqlalchemy.ext.asyncio import AsyncSession
       from app.db.models import AppUser, Category

       class PlanOverflowError(Exception):
           def __init__(self, income_cents: int, sum_plan_cents: int) -> None:
               self.income_cents = income_cents
               self.sum_plan_cents = sum_plan_cents
               super().__init__(
                   f"Σplan ({sum_plan_cents}) exceeds income ({income_cents})"
               )

       class CategoryNotInTenantError(Exception):
           def __init__(self, category_id: int) -> None:
               self.category_id = category_id
               super().__init__(f"Category {category_id} not found")

       async def update_plan_month_atomic(
           db: AsyncSession,
           *,
           user_id: int,
           plans: list[tuple[int, int]],  # (category_id, plan_cents)
       ) -> list[Category]:
           # 1. Pre-validate Σplan ≤ income (если income set).
           income = await db.scalar(
               select(AppUser.income_cents).where(AppUser.id == user_id)
           )
           sum_plan = sum(p[1] for p in plans)
           if income is not None and sum_plan > income:
               raise PlanOverflowError(income, sum_plan)

           # 2. Bulk fetch all referenced categories belonging to user.
           cat_ids = [p[0] for p in plans]
           rows = await db.execute(
               select(Category).where(
                   Category.id.in_(cat_ids),
                   Category.user_id == user_id,
               )
           )
           cats_by_id = {c.id: c for c in rows.scalars().all()}

           # 3. 404 if any requested ID is missing OR cross-tenant.
           for cid in cat_ids:
               if cid not in cats_by_id:
                   raise CategoryNotInTenantError(cid)

           # 4. Apply each plan_cents (in-memory; DB flush at commit-time).
           for cid, pcents in plans:
               cats_by_id[cid].plan_cents = pcents

           await db.flush()
           # Re-fetch in order for stable response (insertion order).
           ordered = [cats_by_id[cid] for cid in cat_ids]
           for c in ordered:
               await db.refresh(c)
           return ordered
       ```
    3. **Создать `app/api/routes/plan_month.py`** (~75 LOC):
       ```python
       """PATCH /api/v1/plan-month — atomic batch plan update (Phase 26, PLAN-V10-06)."""
       from typing import Annotated
       from fastapi import APIRouter, Depends, HTTPException, status
       from sqlalchemy.ext.asyncio import AsyncSession
       from app.api.dependencies import (
           get_current_user, get_current_user_id, get_db_with_tenant_scope,
           require_onboarded,
       )
       from app.api.schemas.categories import CategoryRead
       from app.api.schemas.plan_month import PlanMonthPatch, PlanMonthResponse
       from app.services import plan_month as plan_svc

       plan_month_router = APIRouter(
           prefix="/plan-month",
           tags=["plan-month"],
           dependencies=[Depends(get_current_user), Depends(require_onboarded)],
       )

       @plan_month_router.patch("", response_model=PlanMonthResponse)
       async def patch_plan_month(
           body: PlanMonthPatch,
           db: Annotated[AsyncSession, Depends(get_db_with_tenant_scope)],
           user_id: Annotated[int, Depends(get_current_user_id)],
       ) -> PlanMonthResponse:
           """Atomic batch update of category.plan_cents per PLAN-V10-06."""
           plans = [(p.category_id, p.plan_cents) for p in body.plans]
           try:
               cats = await plan_svc.update_plan_month_atomic(
                   db, user_id=user_id, plans=plans
               )
               await db.commit()
           except plan_svc.PlanOverflowError as exc:
               raise HTTPException(
                   status_code=status.HTTP_400_BAD_REQUEST,
                   detail={
                       "error": "plan_overflow",
                       "income_cents": exc.income_cents,
                       "sum_plan_cents": exc.sum_plan_cents,
                   },
               ) from exc
           except plan_svc.CategoryNotInTenantError as exc:
               raise HTTPException(
                   status_code=status.HTTP_404_NOT_FOUND,
                   detail=f"category {exc.category_id} not found",
               ) from exc
           return PlanMonthResponse(
               categories=[CategoryRead.model_validate(c) for c in cats]
           )
       ```
