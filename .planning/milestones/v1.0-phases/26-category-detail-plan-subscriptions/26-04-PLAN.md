       ```typescript
       import type { CategoryV10, ActualV10Read, SubscriptionV10Read } from '../../api/v10';
       import type { PlanMonthItem } from '../../api/types';

       /** Signed surplus = income − Σplan_cents. Negative means overflow. */
       export function computeSurplus(incomeCents: number, plans: PlanMonthItem[]): number {
         const sum = plans.reduce((s, p) => s + p.plan_cents, 0);
         return incomeCents - sum;
       }

       export function computeIsOverflow(surplusCents: number): boolean {
         return surplusCents < 0;
       }

       export interface RolloverAggregates {
         miscCents: number;
         savingsCents: number;
       }

       /**
        * Aggregate потенциальных rollover'ов:
        * для каждой category где fact ≤ plan, добавить (plan − fact) в bucket по rollover policy.
        * isOver rows вносят 0 (нет остатка).
        */
       export function computeRolloverAggregates(
         categories: CategoryV10[],
         plans: PlanMonthItem[],
         actuals: ActualV10Read[],
       ): RolloverAggregates {
         const planByCat = new Map(plans.map((p) => [p.category_id, p.plan_cents]));
         const factByCat = new Map<number, number>();
         for (const a of actuals) {
           if (a.kind !== 'expense') continue;
           factByCat.set(a.category_id, (factByCat.get(a.category_id) ?? 0) + Math.abs(a.amount_cents));
         }

         let miscCents = 0;
         let savingsCents = 0;
         for (const c of categories) {
           if (c.code === 'savings' || c.paused === true) continue;
           const plan = planByCat.get(c.id) ?? c.plan_cents ?? 0;
           const fact = factByCat.get(c.id) ?? 0;
           const remainder = Math.max(0, plan - fact);
           if (remainder === 0) continue;
           const rollover = c.rollover ?? 'misc';
           if (rollover === 'savings') savingsCents += remainder;
           else miscCents += remainder;
         }
         return { miscCents, savingsCents };
       }

       export interface RegularRow {
         id: number;
         name: string;
         dayOfMonth: number;
         categoryName: string;
         amountCents: number;
         postedTxnId: number | null;
       }

       /**
        * Filter monthly subscriptions с day_of_month set; join category.name.
        * Sort by day_of_month ASC for deterministic display.
        */
       export function computeRegularsList(
         subs: SubscriptionV10Read[],
         categories: CategoryV10[],
       ): RegularRow[] {
         const catName = new Map(categories.map((c) => [c.id, c.name]));
         return subs
           .filter((s) => s.cycle === 'monthly' && s.day_of_month != null)
           .map((s) => ({
             id: s.id,
             name: s.name,
             dayOfMonth: s.day_of_month!,
             categoryName: catName.get(s.category_id) ?? '—',
             amountCents: s.amount_cents,
             postedTxnId: s.posted_txn_id ?? null,
           }))
           .sort((a, b) => a.dayOfMonth - b.dayOfMonth);
       }

       /** Immutable update / add: returns new plans array. */
       export function applyPlanEdit(
         plans: PlanMonthItem[],
         categoryId: number,
         newCents: number,
       ): PlanMonthItem[] {
         const idx = plans.findIndex((p) => p.category_id === categoryId);
         if (idx === -1) {
           return [...plans, { category_id: categoryId, plan_cents: newCents }];
         }
         return plans.map((p, i) =>
           i === idx ? { category_id: p.category_id, plan_cents: newCents } : p,
         );
       }

       /** Initialize plans array from list of categories' current plan_cents. */
       export function plansFromCategories(categories: CategoryV10[]): PlanMonthItem[] {
         return categories
           .filter((c) => c.code !== 'savings' && c.paused !== true)
           .map((c) => ({ category_id: c.id, plan_cents: c.plan_cents ?? 0 }));
       }
       ```
