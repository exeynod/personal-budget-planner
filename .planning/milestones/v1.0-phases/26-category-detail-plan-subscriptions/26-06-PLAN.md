       ```typescript
       import type { SubscriptionV10Read } from '../../api/v10';
       import { MONTHS_RU_GENITIVE } from '../common';

       export function computeActiveCount(subs: SubscriptionV10Read[]): number {
         return subs.filter((s) => s.is_active).length;
       }

       export function computeMonthlyTotal(subs: SubscriptionV10Read[]): number {
         return subs
           .filter((s) => s.is_active && s.cycle === 'monthly')
           .reduce((sum, s) => sum + s.amount_cents, 0);
       }

       export function computeYearlyTotalAnnualized(subs: SubscriptionV10Read[]): number {
         const monthlyAnnual = computeMonthlyTotal(subs) * 12;
         const yearlySum = subs
           .filter((s) => s.is_active && s.cycle === 'yearly')
           .reduce((sum, s) => sum + s.amount_cents, 0);
         return monthlyAnnual + yearlySum;
       }

       /** Human-readable charging cadence in Russian. */
       export function formatCadenceRu(sub: SubscriptionV10Read): string {
         if (sub.cycle === 'monthly') {
           const day = sub.day_of_month;
           if (day != null) return `каждое ${day} число`;
           return 'ежемесячно';
         }
         // yearly — parse next_charge_date «YYYY-MM-DD»
         const d = new Date(sub.next_charge_date);
         if (Number.isNaN(d.getTime())) return 'ежегодно';
         return `${d.getDate()} ${MONTHS_RU_GENITIVE[d.getMonth()]}`;
       }

       /** Sort: active first, then by amount DESC, then by name ASC. */
       export function sortForDisplay(subs: SubscriptionV10Read[]): SubscriptionV10Read[] {
         return [...subs].sort((a, b) => {
           if (a.is_active !== b.is_active) return a.is_active ? -1 : 1;
           if (a.amount_cents !== b.amount_cents) return b.amount_cents - a.amount_cents;
           return a.name.localeCompare(b.name, 'ru');
         });
       }
       ```

       ```typescript
         onClose: () => void;
         onTogglePause: (sub: SubscriptionV10Read) => Promise<void>;
         onChangeDay: (sub: SubscriptionV10Read, newDay: number) => Promise<void>;
         onChangePrice: (sub: SubscriptionV10Read, newAmountCents: number) => Promise<void>;
         onDelete: (sub: SubscriptionV10Read) => Promise<void>;

         const [priceRubles, setPriceRubles] = useState<string>('');

         if (sub === null) return null;

         return (

               onClose={closeAll}

               testId="sub-menu-sheet"
       ```

    1. **Create `SubscriptionsMount.tsx`** (~180 LOC):
       ```typescript
       import { useCallback, useEffect, useState } from 'react';
       import { usePosterRouter } from '../common';
       import {
         listSubscriptionsV10, patchSubscriptionV10, deleteSubscription,
       } from '../../api/v10';
       import type { SubscriptionV10Read } from '../../api/v10';
       import { SubscriptionsView } from './SubscriptionsView';
       import { SubscriptionMenuSheet } from './SubscriptionMenuSheet';

       export function SubscriptionsMount() {
         const router = usePosterRouter();
         const [subs, setSubs] = useState<SubscriptionV10Read[]>([]);
         const [status, setStatus] = useState<'loading'|'ready'|'error'>('loading');
         const [error, setError] = useState('');
         const [menuSub, setMenuSub] = useState<SubscriptionV10Read | null>(null);
         const [reloadToken, setReloadToken] = useState(0);

         useEffect(() => {
           let cancelled = false;
           (async () => {
             try {
               setStatus('loading');
               const list = await listSubscriptionsV10();
               if (cancelled) return;
               setSubs(list);
               setStatus('ready');
             } catch (e) {
               if (cancelled) return;
               setError(e instanceof Error ? e.message : 'Ошибка');
               setStatus('error');
             }
           })();
           return () => { cancelled = true; };
         }, [reloadToken]);

         const refresh = useCallback(() => setReloadToken((n) => n+1), []);

         const handleTogglePause = useCallback(async (sub: SubscriptionV10Read) => {
           try {
             await patchSubscriptionV10(sub.id, { is_active: !sub.is_active });
             refresh();
           } catch { window.alert('Не удалось обновить статус'); }
         }, [refresh]);

         const handleChangeDay = useCallback(async (sub: SubscriptionV10Read, day: number) => {
           try {
             await patchSubscriptionV10(sub.id, { day_of_month: day });
             refresh();
           } catch { window.alert('Не удалось обновить день'); }
         }, [refresh]);

         const handleChangePrice = useCallback(async (sub: SubscriptionV10Read, cents: number) => {
           try {
             await patchSubscriptionV10(sub.id, { amount_cents: cents });
             refresh();
           } catch { window.alert('Не удалось обновить цену'); }
         }, [refresh]);

         const handleDelete = useCallback(async (sub: SubscriptionV10Read) => {
           try {
             await deleteSubscription(sub.id);
             refresh();
           } catch { window.alert('Не удалось удалить подписку'); }
         }, [refresh]);

         if (status === 'loading') return <div data-testid="subs-loading">Загрузка…</div>;
         if (status === 'error') return (
           <div data-testid="subs-error">
             <div>{error}</div>
             <button onClick={refresh}>Повторить</button>
           </div>
         );

         return (
           <>
             <SubscriptionsView
               subs={subs}
               onMenuOpen={(sub) => setMenuSub(sub)}
               onBack={() => router.pop()}
             />
             <SubscriptionMenuSheet
               sub={menuSub}
               onClose={() => setMenuSub(null)}
               onTogglePause={handleTogglePause}
               onChangeDay={handleChangeDay}
               onChangePrice={handleChangePrice}
               onDelete={handleDelete}
             />
           </>
         );
       ```
