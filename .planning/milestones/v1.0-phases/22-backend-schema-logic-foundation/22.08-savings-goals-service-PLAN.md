"""Savings service (BE-08 config, BE-09 aggregator, BE-10 deposit).

Module contents:
    get_savings_snapshot — read-side aggregator (DATA-MODEL §2.4).
    upsert_config         — write-side INSERT … ON CONFLICT for SavingsConfig.
    create_deposit        — calls create_actual_v10 with kind=deposit + optional Goal bump.
"""
from datetime import datetime
from zoneinfo import ZoneInfo

from sqlalchemy import func, select
from sqlalchemy.dialects.postgresql import insert as pg_insert
from sqlalchemy.ext.asyncio import AsyncSession

from app.db.models import (
    ActualKind, ActualSource, ActualTransaction, Category, Goal, SavingsConfig
)


_DEFAULT_CONFIG = {"roundup_enabled": False, "roundup_base": 10}
_VALID_BASES = (10, 50, 100)


def _msk_month_bounds():
    tz = ZoneInfo("Europe/Moscow")
    today = datetime.now(tz).date()
    first_of_month = today.replace(day=1)
    return first_of_month  # tx_date >= first_of_month → in current MSK month


async def get_savings_snapshot(db: AsyncSession, *, user_id: int) -> dict:
    """BE-09 aggregator: {total_cents, month_in_cents, config, goals[]}."""
    # Total: SUM(ABS(amount_cents)) for roundup + deposit
    total_q = select(func.coalesce(func.sum(func.abs(ActualTransaction.amount_cents)), 0)).where(
        ActualTransaction.user_id == user_id,
        ActualTransaction.kind.in_([ActualKind.roundup, ActualKind.deposit]),
    )
    total = (await db.execute(total_q)).scalar_one()

    # Month-in: same filter + tx_date >= first_of_month MSK
    month_start = _msk_month_bounds()
    month_q = select(func.coalesce(func.sum(func.abs(ActualTransaction.amount_cents)), 0)).where(
        ActualTransaction.user_id == user_id,
        ActualTransaction.kind.in_([ActualKind.roundup, ActualKind.deposit]),
        ActualTransaction.tx_date >= month_start,
    )
    month_in = (await db.execute(month_q)).scalar_one()

    cfg = await db.scalar(select(SavingsConfig).where(SavingsConfig.user_id == user_id))
    if cfg is None:
        config_dict = _DEFAULT_CONFIG.copy()
    else:
        config_dict = {
            "roundup_enabled": cfg.roundup_enabled,
            "roundup_base": cfg.roundup_base,
        }

    goals = (await db.execute(
        select(Goal).where(Goal.user_id == user_id).order_by(Goal.created_at)
    )).scalars().all()

    return {
        "total_cents": int(total),
        "month_in_cents": int(month_in),
        "config": config_dict,
        "goals": [
            {
                "id": g.id, "name": g.name,
                "target_cents": g.target_cents, "current_cents": g.current_cents,
                "due": g.due.isoformat() if g.due else None,
                "created_at": g.created_at.isoformat(),
            }
            for g in goals
        ],
    }


async def upsert_config(
    db: AsyncSession, *, user_id: int,
    roundup_enabled: bool | None = None, roundup_base: int | None = None,
) -> SavingsConfig:
    """BE-08 PATCH /savings/config — INSERT or UPDATE (partial)."""
    if roundup_base is not None and roundup_base not in _VALID_BASES:
        raise ValueError(f"roundup_base must be in {_VALID_BASES}; got {roundup_base}")

    # Build values dict — only include explicitly provided fields
    values = {"user_id": user_id}
    if roundup_enabled is not None:
        values["roundup_enabled"] = roundup_enabled
    if roundup_base is not None:
        values["roundup_base"] = roundup_base

    update_set = {k: v for k, v in values.items() if k != "user_id"}
    update_set["updated_at"] = func.now()

    if not update_set or (len(update_set) == 1 and "updated_at" in update_set):
        # No-op patch — just return existing
        existing = await db.scalar(select(SavingsConfig).where(SavingsConfig.user_id == user_id))
        if existing:
            return existing
        # Else fall through to INSERT defaults

    stmt = (
        pg_insert(SavingsConfig)
        .values(**values)
        .on_conflict_do_update(index_elements=["user_id"], set_=update_set)
        .returning(SavingsConfig)
    )
    result = await db.execute(stmt)
    await db.flush()
    cfg = result.scalar_one()
    return cfg


async def create_deposit(
    db: AsyncSession, *, user_id: int,
    amount_cents: int, account_id: int, goal_id: int | None = None,
) -> ActualTransaction:
    """BE-10 POST /savings/deposit — kind=deposit txn + optional Goal bump."""
    if amount_cents == 0:
        raise ValueError("amount_cents must be non-zero")
    # Convention: deposit reduces account balance; amount stored negative
    if amount_cents > 0:
        amount_cents = -amount_cents

    savings_cat = await db.scalar(
        select(Category).where(Category.user_id == user_id, Category.code == "savings")
    )
    if savings_cat is None:
        from app.services.roundup import SavingsCategoryMissingError
        raise SavingsCategoryMissingError(user_id)

    if goal_id is not None:
        goal = await db.scalar(
            select(Goal).where(Goal.id == goal_id, Goal.user_id == user_id)
        )
        if goal is None:
            from app.services.goals import GoalNotFoundError
            raise GoalNotFoundError(goal_id)

    from app.services.actual import create_actual_v10
    from app.services.periods import _today_in_app_tz

    parent, _ = await create_actual_v10(
        db,
        user_id=user_id,
        kind="deposit",
        amount_cents=amount_cents,
        description="Пополнение копилки",
        category_id=savings_cat.id,
        tx_date=_today_in_app_tz(),
        source=ActualSource.mini_app,
        account_id=account_id,
    )

    if goal_id is not None:
        # Bump Goal.current_cents atomically in same transaction
        await db.execute(
            text("UPDATE goal SET current_cents = current_cents + :amt "
                 "WHERE id = :gid AND user_id = :uid"),
            {"amt": abs(amount_cents), "gid": goal_id, "uid": user_id}
        )

    return parent
```

"""Goal CRUD service (BE-11). Validators per DATA-MODEL §6."""
from datetime import date as date_type
from typing import Optional

from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.db.models import Goal
from app.services.periods import _today_in_app_tz


class GoalNotFoundError(Exception):
    def __init__(self, goal_id: int) -> None:
        self.goal_id = goal_id
        super().__init__(f"Goal {goal_id} not found")


def _validate_name(name: str) -> None:
    if not name or len(name) > 80:
        raise ValueError(f"Goal name length must be 1..80; got {len(name)}")


def _validate_target(target_cents: int) -> None:
    if target_cents <= 0:
        raise ValueError(f"Goal target_cents must be > 0; got {target_cents}")


def _validate_due(due: Optional[date_type]) -> None:
    if due is None:
        return
    if due <= _today_in_app_tz():
        raise ValueError(f"Goal due must be in the future; got {due}")


async def list_goals(db: AsyncSession, *, user_id: int) -> list[Goal]:
    return list((await db.execute(
        select(Goal).where(Goal.user_id == user_id).order_by(Goal.created_at)
    )).scalars().all())


async def get_or_404(db: AsyncSession, goal_id: int, *, user_id: int) -> Goal:
    row = await db.scalar(
        select(Goal).where(Goal.id == goal_id, Goal.user_id == user_id)
    )
    if row is None:
        raise GoalNotFoundError(goal_id)
    return row


async def create_goal(
    db: AsyncSession, *, user_id: int,
    name: str, target_cents: int, due: Optional[date_type] = None,
) -> Goal:
    _validate_name(name)
    _validate_target(target_cents)
    _validate_due(due)
    row = Goal(
        user_id=user_id, name=name,
        target_cents=target_cents, current_cents=0, due=due,
    )
    db.add(row)
    await db.flush()
    await db.refresh(row)
    return row


async def update_goal(
    db: AsyncSession, goal_id: int, *, user_id: int, patch: dict,
) -> Goal:
    row = await get_or_404(db, goal_id, user_id=user_id)
    if "name" in patch:
        _validate_name(patch["name"])
    if "target_cents" in patch:
        _validate_target(patch["target_cents"])
    if "due" in patch:
        _validate_due(patch["due"])
    for k, v in patch.items():
        setattr(row, k, v)
    await db.flush()
    await db.refresh(row)
    return row


async def delete_goal(db: AsyncSession, goal_id: int, *, user_id: int) -> Goal:
    row = await get_or_404(db, goal_id, user_id=user_id)
    await db.delete(row)
    await db.flush()
    return row
```
