```python
class SubscriptionAlreadyPostedError(Exception):
    """BE-13 idempotency: subscription already has posted_txn_id linked."""
    def __init__(self, sub_id: int, posted_txn_id: int) -> None:
        self.sub_id = sub_id
        self.posted_txn_id = posted_txn_id
        super().__init__(f"Subscription {sub_id} already posted (txn {posted_txn_id})")


class SubscriptionNotPostedError(Exception):
    """BE-13 unpost guard: nothing to unpost."""
    def __init__(self, sub_id: int) -> None:
        self.sub_id = sub_id
        super().__init__(f"Subscription {sub_id} is not posted")


class SubscriptionInactiveError(Exception):
    def __init__(self, sub_id: int) -> None:
        self.sub_id = sub_id
        super().__init__(f"Subscription {sub_id} is inactive")


async def post_subscription(
    db: AsyncSession, sub_id: int, *, user_id: int,
) -> ActualTransaction:
    """BE-13 POST /subscriptions/:id/post.

    Creates expense actual_transaction tied to subscription's category and
    account; sets subscription.posted_txn_id = new txn.id; applies balance delta.

    Idempotent failure: returns 409 if already posted (caller maps).
    """
    from app.services.actual import create_actual_v10
    from app.services.periods import _today_in_app_tz

    sub = await db.scalar(
        select(Subscription).where(
            Subscription.id == sub_id,
            Subscription.user_id == user_id,
        )
    )
    if sub is None:
        # Reuse existing SubscriptionNotFoundError if defined; else raise generic
        raise SubscriptionNotFoundError(sub_id)

    if sub.posted_txn_id is not None:
        raise SubscriptionAlreadyPostedError(sub_id, sub.posted_txn_id)

    if not sub.is_active:
        raise SubscriptionInactiveError(sub_id)

    if sub.account_id is None:
        raise ValueError(
            f"Subscription {sub_id} has no account_id — cannot post without account"
        )

    # Subscription.amount_cents is positive in DB convention; expense txn must be negative
    txn_amount = -abs(sub.amount_cents)

    parent, _child = await create_actual_v10(
        db,
        user_id=user_id,
        kind="expense",
        amount_cents=txn_amount,
        description=f"Подписка: {sub.name}",
        category_id=sub.category_id,
        tx_date=_today_in_app_tz(),
        source=ActualSource.mini_app,
        account_id=sub.account_id,
    )

    sub.posted_txn_id = parent.id
    await db.flush()
    return parent


async def unpost_subscription(
    db: AsyncSession, sub_id: int, *, user_id: int,
) -> None:
    """BE-13 POST /subscriptions/:id/unpost.

    Deletes the linked actual_transaction (CASCADE removes any roundup child)
    and clears subscription.posted_txn_id. Account balance restored by
    delete_actual_v10.
    """
    from app.services.actual import delete_actual_v10

    sub = await db.scalar(
        select(Subscription).where(
            Subscription.id == sub_id,
            Subscription.user_id == user_id,
        )
    )
    if sub is None:
        raise SubscriptionNotFoundError(sub_id)

    if sub.posted_txn_id is None:
        raise SubscriptionNotPostedError(sub_id)

    txn_id = sub.posted_txn_id
    sub.posted_txn_id = None  # clear before delete to avoid FK churn
    await db.flush()

    await delete_actual_v10(db, txn_id, user_id=user_id)
```
