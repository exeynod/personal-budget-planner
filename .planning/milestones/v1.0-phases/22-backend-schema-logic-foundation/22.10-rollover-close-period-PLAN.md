```python
"""Rollover service (BE-14).

Called from close_period_job between compute_balance and status flip.
Implements DATA-MODEL §3 rollover rules with CONTEXT §Area 3 idempotency.

Trigger: do_period_rollover(session, *, period_id, user_id, next_period_id=None)
  Acquires pg_try_advisory_lock(hashtext('close_period:' || period_id)).
  Returns False if lock contention; True on successful (or already-processed) run.
  Inside lock: if period.rollover_processed_at IS NOT NULL → return True (no-op).
  Else: iterate non-paused categories, branch on rollover policy, set
  rollover_processed_at = now(), UPDATE next_period.misc_rollover_cents.
"""
from __future__ import annotations

from datetime import datetime, timezone
from typing import Optional

import structlog
from sqlalchemy import func, select, text
from sqlalchemy.ext.asyncio import AsyncSession

from app.db.models import (
    Account, ActualKind, ActualSource, ActualTransaction,
    BudgetPeriod, Category, RolloverPolicy
)

logger = structlog.get_logger(__name__)


class RolloverConfigError(Exception):
    """Misconfiguration blocking rollover (e.g. no primary account)."""


async def _try_lock(session: AsyncSession, period_id: int) -> bool:
    """pg_try_advisory_lock keyed on period_id; True if acquired."""
    result = await session.execute(
        text("SELECT pg_try_advisory_xact_lock(hashtext(:k))"),
        {"k": f"close_period:{period_id}"},
    )
    return bool(result.scalar())


async def do_period_rollover(
    session: AsyncSession, *,
    period_id: int,
    user_id: int,
    next_period_id: Optional[int] = None,
) -> bool:
    """BE-14 rollover. Returns True on success or already-done; False on lock contention.

    Caller must have an open transaction. We use pg_try_advisory_xact_lock so
    the lock auto-releases on COMMIT/ROLLBACK.
    """
    if not await _try_lock(session, period_id):
        logger.info("rollover.skipped.lock_contention", period_id=period_id)
        return False

    period = await session.scalar(
        select(BudgetPeriod).where(
            BudgetPeriod.id == period_id, BudgetPeriod.user_id == user_id
        )
    )
    if period is None:
        logger.warning("rollover.period_not_found", period_id=period_id, user_id=user_id)
        return True  # Nothing to do
    if period.rollover_processed_at is not None:
        logger.info("rollover.already_processed", period_id=period_id)
        return True

    # Look up primary account (fall back to None if user has no accounts)
    primary = await session.scalar(
        select(Account).where(
            Account.user_id == user_id,
            Account.is_primary.is_(True),
        )
    )

    # Find savings category
    savings_cat = await session.scalar(
        select(Category).where(
            Category.user_id == user_id,
            Category.code == "savings",
        )
    )

    # Iterate non-paused, non-savings categories
    cats_q = select(Category).where(
        Category.user_id == user_id,
        Category.paused.is_(False),
        Category.is_archived.is_(False),
    )
    cats = (await session.execute(cats_q)).scalars().all()

    misc_total = 0
    for cat in cats:
        if cat.code == "savings":
            continue  # never rollover the savings cat into itself
        # Aggregate fact for this category in the period (kind='expense' only)
        fact = await session.scalar(
            select(func.coalesce(func.sum(func.abs(ActualTransaction.amount_cents)), 0))
            .where(
                ActualTransaction.user_id == user_id,
                ActualTransaction.period_id == period_id,
                ActualTransaction.category_id == cat.id,
                ActualTransaction.kind == ActualKind.expense,
            )
        )
        remainder = max(0, (cat.plan_cents or 0) - int(fact or 0))
        if remainder == 0:
            continue

        if cat.rollover == RolloverPolicy.savings:
            if savings_cat is None:
                from app.services.roundup import SavingsCategoryMissingError
                raise SavingsCategoryMissingError(user_id)
            if primary is None:
                raise RolloverConfigError(
                    f"User {user_id} has no primary account; cannot create rollover deposit"
                )
            txn = ActualTransaction(
                user_id=user_id,
                period_id=period_id,
                kind=ActualKind.deposit,
                amount_cents=-remainder,
                description=f"Остаток {cat.name} → копилка",
                category_id=savings_cat.id,
                tx_date=period.period_end,
                source=ActualSource.mini_app,  # synthetic — closest match
                account_id=primary.id,
            )
            session.add(txn)
            await session.flush()
            # Apply balance delta on primary account
            from app.services.accounts import apply_balance_delta
            await apply_balance_delta(
                session, primary.id, user_id=user_id, delta_cents=-remainder
            )
        else:  # misc
            misc_total += remainder

    # Apply misc accumulation to next period if provided
    if misc_total > 0 and next_period_id is not None:
        await session.execute(
            text(
                "UPDATE budget_period SET misc_rollover_cents = misc_rollover_cents + :amt "
                "WHERE id = :pid AND user_id = :uid"
            ),
            {"amt": misc_total, "pid": next_period_id, "uid": user_id},
        )

    # Mark processed
    period.rollover_processed_at = datetime.now(timezone.utc)
    await session.flush()

    logger.info(
        "rollover.done",
        user_id=user_id, period_id=period_id,
        next_period_id=next_period_id,
        misc_carry_cents=misc_total,
    )
    return True
```
