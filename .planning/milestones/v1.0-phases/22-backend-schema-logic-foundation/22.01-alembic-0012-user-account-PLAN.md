    "account",
    sa.Column("id", sa.BigInteger(), primary_key=True, autoincrement=True),

    sa.Column(

        sa.Enum("card", "cash", "savings", name="account_kind", create_type=True),
        nullable=False,
    ),
    sa.Column("balance_cents", sa.BigInteger(), nullable=False, server_default="0"),
    sa.Column("primary", sa.Boolean(), nullable=False, server_default=sa.text("false")),
    sa.Column(
        "created_at",
        sa.TIMESTAMP(timezone=True),
        server_default=sa.func.now(),
        nullable=False,
    ),
    sa.CheckConstraint("char_length(bank) BETWEEN 1 AND 40", name="ck_account_bank_length"),
    sa.CheckConstraint("balance_cents >= -100000000000 AND balance_cents <= 100000000000", name="ck_account_balance_range"),
